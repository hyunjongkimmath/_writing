\begin{definition} \label{definition:index_of_a_subgroup_in_a_group}
    Let $G$ be a \CrefAndHyperrefIfExist{definition:group}{group} and let $H \leq G$ be a \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup}.
    The \hldef{index of $H$ in $G$} is the \CrefAndHyperrefIfExist{definition:cardinal_number_and_cardinality_of_a_set}{cardinality} of the set 
    $$\{gH: g \in G\}$$
    of left cosets of $H$ in $G$. This is equivalent to the cardinality of the set 
    $$\{Hg: g \in G\}$$
    of right cosets of $H$ in $G$.
\end{definition}

\begin{definition} \label{definition:kernel_image_of_group_homomorphism}
Let $\phi: G \to H$ be a \CrefAndHyperrefIfExist{definition:group_homomorphism}{group homomorphism}.
\begin{enumerate}
    \item The \hldef{kernel of $\phi$}, denoted \hl{$\operatorname{ker}(\phi)$}, is the set of elements in $G$ that map to the identity element of $H$:
    $$ \operatorname{ker}(\phi) = \{ g \in G \mid \phi(g) = e_H \} $$
    where $e_H$ is the identity element of $H$. 

    \item The \hldef{image of $\phi$,} denoted \hl{$\operatorname{im}(\phi)$} or \hl{$\phi(G)$}, is the set of elements in $H$ that are the image of some element in $G$:
    $$ \operatorname{im}(\phi) = \{ h \in H \mid \exists g \in G, \phi(g) = h \} $$
\end{enumerate}
\end{definition}


\begin{proposition} \label{proposition:basic_facts_about_kernel_and_image_of_group_homomorphisms}
Let $\phi: G \to H$ be a \CrefAndHyperrefIfExist{definition:group_homomorphism}{group homomorphism}.
\begin{enumerate}
    \item The \CrefAndHyperrefIfExist{definition:kernel_image_of_group_homomorphism}{kernel $\operatorname{ker}(\phi)$} is a \CrefAndHyperrefIfExist{definition:normal_subgroup_of_a_group}{normal subgroup} of $G$.
    \item The \CrefAndHyperrefIfExist{definition:kernel_image_of_group_homomorphism}{image} $\operatorname{im}(\phi)$ is a \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup} of $H$.
    \item The homomorphism $\phi$ is \CrefAndHyperrefIfExist{definition:injective_surjective_bijective_map_of_sets}{injective} if and only if $\operatorname{ker}(\phi) = \{e_G\}$, where $e_G$ is the identity element of $G$.
    \item The homomorphism $\phi$ is \CrefAndHyperrefIfExist{definition:injective_surjective_bijective_map_of_sets}{surjective} if and only if $\operatorname{im}(\phi) = H$.
\end{enumerate}
\end{proposition}

\begin{proposition} \label{proposition:quotient_group_of_a_group_by_a_normal_subgroup_has_well_deinfed_operation}
Let $G$ be a \CrefAndHyperrefIfExist{definition:group}{group} and $N$ a \CrefAndHyperrefIfExist{definition:normal_subgroup_of_a_group}{normal subgroup} of $G$.
The binary operation of the \CrefAndHyperrefIfExist{definition:quotient_group_of_a_group_by_a_normal_subgroup}{quotient group} $G/N$ is well defined and $G/N$ is indeed a group.
\end{proposition}

\begin{theorem} \label{theorem:canonical_projection_from_group_to_quotient_by_normal_subgroup}
The map $\pi: G \to G/N$ defined by $\pi(g) = gN$ is a \CrefAndHyperrefIfExist{definition:injective_surjective_bijective_map_of_sets}{surjective} \CrefAndHyperrefIfExist{definition:group_homomorphism}{group homomorphism} with \CrefAndHyperrefIfExist{definition:kernel_image_of_group_homomorphism}{kernel} $N$. This map is called the \hldef{canonical projection} or \hldef{natural homomorphism}.
\end{theorem}



\begin{theorem}
    \begin{enumerate}
        \item (First Isomorphism Theorem) Let $\phi: G \to H$ be a group homomorphism with kernel $K = \operatorname{ker}(\phi)$ and image $I = \operatorname{im}(\phi)$. Then $K$ is a normal subgroup of $G$, $I$ is a subgroup of $H$, and there is an isomorphism
        $$ G/K \cong I $$
        given by the map $gK \mapsto \phi(g)$. This theorem implies that any homomorphic image of a group is isomorphic to a quotient of that group.

        \item (Second Isomorphism Theorem, or Diamond Isomorphism Theorem) Let $G$ be a group, $H$ a subgroup of $G$, and $N$ a normal subgroup of $G$. Then:
        1. The product $HN = \{hn \mid h \in H, n \in N\}$ is a subgroup of $G$.
        2. The intersection $H \cap N$ is a normal subgroup of $H$.
        3. There is an isomorphism
        $$ H / (H \cap N) \cong (HN) / N $$
        given by the map $h(H \cap N) \mapsto hN$.

        \item (Third Isomorphism Theorem) Let $G$ be a group and let $N$ and $K$ be normal subgroups of $G$ with $N \subseteq K$. Then $K/N$ is a normal subgroup of $G/N$, and there is an isomorphism
        $$ (G/N) / (K/N) \cong G/K $$
        given by the map $(gN)(K/N) \mapsto gK$.

        \item (Correspondence Theorem, or Lattice Isomorphism Theorem) Let $G$ be a group and $N$ be a normal subgroup of $G$. There is a one-to-one correspondence between the set of subgroups $A$ of $G$ containing $N$ (i.e., $N \subseteq A \subseteq G$) and the set of subgroups of the quotient group $G/N$. This correspondence is given by $A \mapsto A/N$, and it preserves properties such as inclusion, indices, normality, and intersections.
    \end{enumerate}
\end{theorem}

\begin{proposition}
    Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} \CrefAndHyperrefIfExist{definition:left_and_right_group_actions_on_a_set}{acting} (on the left) on a set $X$.    
    Define a \CrefAndHyperrefIfExist{definition:n_ary_relation_on_sets}{binary relation} $\sim$ on $X$ by
    $$x \sim y \;\; \Longleftrightarrow \;\; \exists g \in G : g \cdot x = y.$$
    Then $\sim$ is an \CrefAndHyperrefIfExist{definition:equivalence_relation_on_a_set}{equivalence relation} on $X$, and the \CrefAndHyperrefIfExist{definition:equivalence_class_of_an_element_of_a_set_under_an_equivalence_relation_and_quotient_of_set_by_equivalence_relation}{equivalence classes} are exactly the \CrefAndHyperrefIfExist{definition:orbit_of_a_group_action}{orbits} $\{ G \cdot x : x \in X \}$.


\end{proposition}

\begin{proposition}
    Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} \CrefAndHyperrefIfExist{definition:left_and_right_group_actions_on_a_set}{acting} (on the left) on a set $X$.    
    For all $x \in X$, the \CrefAndHyperrefIfExist{definition:stabilizer_of_an_element_of_a_set_under_a_group_action}{stabilizer}
        $$\operatorname{Stab}_G(x) = \{ g \in G : g \cdot x = x \}$$
        is a \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup} of $G$.
\end{proposition}

\begin{proposition}
    Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} \CrefAndHyperrefIfExist{definition:left_and_right_group_actions_on_a_set}{acting} (on the left) on a set $X$.    
    For all $x \in X$, there is a bijection of sets
    $$G / \operatorname{Stab}_G(x) \;\;\longleftrightarrow\;\; G \cdot x, \quad g \operatorname{Stab}_G(x) \mapsto g \cdot x.$$
    In particular, if $G$ is finite then
    $$|G \cdot x| = [G : \operatorname{Stab}_G(x)] = \frac{|G|}{|\operatorname{Stab}_G(x)|}.$$

\end{proposition}

\begin{proposition}
    Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} \CrefAndHyperrefIfExist{definition:left_and_right_group_actions_on_a_set}{acting} (on the left) on a set $X$.    
    For any subgroup $H \subseteq G$, the \CrefAndHyperrefIfExist{definition:fixed_point_set_of_a_group_acting_on_a_set}{fixed point set}
    $$X^H = \{ x \in X : h \cdot x = x \text{ for all } h \in H \}$$
    satisfies $X^H \subseteq X^{H'}$ whenever $H' \subseteq H$.
\end{proposition}

\subsection{Short exact sequences of general groups}


\begin{definition} \label{definition:exact_sequence_short_exact_sequence_of_group_homomorphisms}
    \begin{enumerate}
        \item 
        A sequence of \CrefAndHyperrefIfExist{definition:group}{groups} and \CrefAndHyperrefIfExist{definition:group_homomorphism}{group homomorphisms}
        $$ \dots \xrightarrow{\phi_{i-1}} G_i \xrightarrow{\phi_i} G_{i+1} \xrightarrow{\phi_{i+1}} \dots $$
        is called \hldef{exact at $G_i$} if \CrefAndHyperrefIfExist{definition:kernel_image_of_group_homomorphism}{$\operatorname{im}(\phi_{i-1}) = \operatorname{ker}(\phi_i)$}. The sequence is called an \hldef{exact sequence} if it is exact at every group $G_i$.

        \item A \hldef{short exact sequence of groups} is an exact sequence of the form
        $$ 1 \longrightarrow N \xrightarrow{\iota} G \xrightarrow{\pi} H \longrightarrow 1 $$
        where $1$ denotes the trivial group. Exactness implies that $\iota$ is \CrefAndHyperrefIfExist{definition:injective_surjective_bijective_map_of_sets}{injective}, $\pi$ is \CrefAndHyperrefIfExist{definition:injective_surjective_bijective_map_of_sets}{surjective}, and $\operatorname{im}(\iota) = \operatorname{ker}(\pi)$.
    \end{enumerate}
\end{definition}


\begin{definition} \label{definition:section_and_splitting_of_a_short_exact_sequence_of_groups}
Let $1 \longrightarrow N \xrightarrow{\iota} G \xrightarrow{\pi} H \longrightarrow 1$ be a short exact sequence of groups. 
\begin{enumerate}
    \item A \hldef{section} (or \hldef{splitting homomorphism)} of the short exact sequence is a \CrefAndHyperrefIfExist{definition:group_homomorphism}{group homomorphism} $s: H \to G$ such that $\pi \circ s = \operatorname{id}_H$.

    \item The short exact sequence $1 \longrightarrow N \xrightarrow{\iota} G \xrightarrow{\pi} H \longrightarrow 1$ is said to \hldef{split} if there exists a section $s: H \to G$.
\end{enumerate}
\end{definition}

\begin{theorem}[Adjunction for Rings]
    \TODO{}
Let $f: R \to S$ be a homomorphism of rings.
Let $\mathsf{Ring}_R$ and $\mathsf{Ring}_S$ denote the \CrefAndHyperrefIfExist{definition:category_of_rings_over_a_ring}{categories of $R$-rings} and $S$-rings, respectively.

There is an adjunction between these categories:
\begin{enumerate}
    \item The \hldef{restriction of scalars functor}, $f_*: \mathsf{Ring}_S \to \mathsf{Ring}_R$, sends an $S$-ring $(B, \psi: S \to B)$ to the $R$-ring $(B, \psi \circ f)$.
    \item The \hldef{extension of scalars functor}, $f^*: \mathsf{Ring}_R \to \mathsf{Ring}_S$, sends an $R$-ring $(A, \varphi: R \to A)$ to the \hldef{pushout} $S \amalg_R A$ in the category of rings (the free product of $S$ and $A$ with amalgamation over $R$).
\end{enumerate}

Then $f^*$ is left adjoint to $f_*$. For any $R$-ring $A$ and any $S$-ring $B$, there is a natural bijection:
\[
    \hlin{ \Hom_{\mathsf{Ring}_S}(S \amalg_R A, B) \cong \Hom_{\mathsf{Ring}_R}(A, f_*B) }.
\]

\textbf{Special Case (Algebras):}
If $R$ is commutative and we restrict to the full subcategory $\mathsf{Alg}_R$ of $R$-algebras (where the structure map maps to the center), and similarly for $S$ (assuming $S$ is an $R$-algebra), then the extension functor is given by the tensor product $S \otimes_R A$, and the adjunction becomes:
\[
    \hlin{ \Hom_{\mathsf{Alg}_S}(S \otimes_R A, B) \cong \Hom_{\mathsf{Alg}_R}(A, f_*B) }.
\]
\end{theorem}
