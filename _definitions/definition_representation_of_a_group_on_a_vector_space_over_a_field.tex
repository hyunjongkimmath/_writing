\begin{definition} \label{definition:representation_of_a_group_on_a_vector_space_over_a_field}
    Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} and $(\mathrm{GL}(V),\circ)$ the group of invertible linear transformations on a vector space $V$ over a field $k$.  
    A \hldef{representation of $G$ on $V$} is a \CrefAndHyperrefIfExist{definition:group_homomorphism}{group homomorphism} 
    $$ \rho : G \to \mathrm{GL}(V). $$
    A pair $(\rho, V)$ is often referred to as a \hldef{representation of $G$}.

    \TextIfExists{definition:representation_of_a_group_on_a_module_over_a_not_necessarily_commutative_ring}{Equivalently, a representation of $G$ on $V$ is a \CrefAndHyperrefIfExist{definition:representation_of_a_group_on_a_module_over_a_not_necessarily_commutative_ring}{representation of $G$} on the \CrefAndHyperrefIfExist{definition:module_of_a_ring}{$k$-module} $V$.}\TODO{state that a $k$-module is the same as a $k$-vector space and cref here}
\end{definition}
