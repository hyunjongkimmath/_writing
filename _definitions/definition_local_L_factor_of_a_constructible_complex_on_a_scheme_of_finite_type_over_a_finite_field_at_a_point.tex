
\begin{definition} \label{definition:local_L_factor_of_a_constructible_complex_on_a_scheme_of_finite_type_over_a_finite_field_at_a_point}
Let \(X_0\) be a scheme of finite type over the finite field \(\mathbb{F}_q\), let $\ell \neq \Char \Fq$ be a prime, and let \(K_0 \in D^b_c(X_0, \overline{\mathbb{Q}}_\ell)\).

The \hldef{local factor of the $L$-function at $x$} is defined as

$$\hlin{L_x(K_0, t) := \det\left( 1 - t^{d(x)} \,\mathrm{Frob}_x \mid H^*(K_{0, \bar{x}}) \right)^{-1},}$$

\CrefIfExists{definition:determinant_over_total_cohomology_of_a_complex} where the determinant is taken over the total cohomology of the stalk at \(\bar{x}\). More explicitly,

$$L_x(K_0, t) := \prod_{i \in \mathbb{Z}} \det\left( 1 - t^{d(x)} \,\mathrm{Frob}_x \mid H^i(K_{0, \bar{x}}) \right)^{(-1)^{i+1}}.$$
\end{definition}
