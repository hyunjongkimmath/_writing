\begin{definition} \label{definition:prime_and_weil_divisor_on_a_scheme}
Let $X$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme}. 
\begin{enumerate}
    \item A \hldef{prime divisor on $X$} is an integral \CrefAndHyperrefIfExist{definition:open_and_closed_subscheme_of_a_scheme}{closed subscheme} $Y \subset X$ of \CrefAndHyperrefIfExist{definition:codimension_of_an_irreducible_closed_subset_in_a_topological_space}{codimension} one. 
    \item A \hldef{Weil divisor} (or simply a \hldef{divisor}) on $X$ is a formal linear combination $D = \sum_{i} n_i Y_i$, where the $Y_i$ are prime divisors and the coefficients $n_i$ are integers, with only finitely many $n_i$ non-zero.
\end{enumerate}

$X$ is often assumed to be \CrefAndHyperrefIfExist{definition:locally_noetherian_and_noetherian_scheme}{locally Noetherian} to ensure nice properties of Weil divisors.
\end{definition}
