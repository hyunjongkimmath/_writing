\begin{definition} \label{definition:morphism_of_sheaves_with_a_property_of_morphisms_of_S_schemes_preserved_under_base_change_and_tau_local_on_the_base}
    Let $S$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme}. Let $\tau$ be some \CrefAndHyperrefIfExist{definition:grothendieck_topology_on_a_category_site_covering_sieve_topologically_generating_family}{Grothendieck topology} on \CrefAndHyperrefIfExist{definition:scheme_over_a_scheme}{$\textit{Sch}/S$}. Let $\mathcal{P}$ be a property of morphisms of $S$-schemes that is preserved under \CrefAndHyperrefIfExist{definition:base_change_of_a_morphism_in_a_category_by_a_morphism}{base change} and \CrefAndHyperrefIfExist{definition:tau_local_property_of_a_morphism_of_schemes_over_a_scheme}{$\tau$-local on the base}. Let $a: F \to G$ be a morphism of $\tau$-sheaves that is \CrefAndHyperrefIfExist{definition:representable_by_schemes_for_a_morphism_of_presheaves_on_Sch_S}{representable by schemes}.
    We say that $a$ \hldef{has property $\mathcal{P}$} if for every scheme $T$ over $S$ and any morphism $h_T \to G$ of sheaves, the morphism $h_T \times_G F \to T$ of schemes, arising by Definition \ref{definition:representable_by_schemes_for_a_morphism_of_presheaves_on_Sch_S}, has property $\mathcal{P}$.
\end{definition}
