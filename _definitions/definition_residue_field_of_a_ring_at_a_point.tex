\begin{definition} \label{definition:residue_field_of_a_ring_at_a_point}
    \begin{enumerate}
        \item Let $R$ be a \CrefAndHyperrefIfExist{definition:local_ring}{local ring} with a unique \CrefAndHyperrefIfExist{definition:prime_and_maximal_ideal_of_a_ring}{maximal ideal} $\mathfrak{m} \subseteq R$. The \hldef{residue field of $R$} is the field
        $$\hlin{\kappa(R) := R/\mathfrak{m}.}$$


        \item Let $R$ be a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative ring} and let $\mathfrak{p} \subseteq R$ be a \CrefAndHyperrefIfExist{definition:prime_and_maximal_ideal_of_a_ring}{prime ideal}. The \hldef{residue field of $R$ at $\mathfrak{p}$} is the field
        $$\hlin{\kappa(\mathfrak{p}) := \mathrm{Frac}(R_{\mathfrak{p}}/\mathfrak{p} R_{\mathfrak{p}}),}$$
        \CrefIfExists{definition:field_of_fractions_of_an_integral_domain}
        where $R_{\mathfrak{p}}$ is the \CrefAndHyperrefIfExist{definition:localization_of_a_commutative_ring_by_a_multiplicative_subset}{localization} of $R$ at the prime $\mathfrak{p}$.
    \end{enumerate}
\end{definition}
