\begin{definition}[Center of a ring] \label{definition:center_of_a_ring}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{(not necessarily commutative) ring}. The \hldef{center of $R$}, denoted by \hl{$Z(R)$}, is the subset of elements in $R$ that commute with every element of $R$:
    $$Z(R) := \{ z \in R \mid zr = rz \text{ for all } r \in R \}.$$
    The center $Z(R)$ is a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative} \CrefAndHyperrefIfExist{definition:subring_of_a_ring}{subring} of $R$.
\end{definition}
