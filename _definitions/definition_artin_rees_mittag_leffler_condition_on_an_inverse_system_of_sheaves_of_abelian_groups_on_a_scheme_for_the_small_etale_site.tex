\begin{definition}[See e.g. {\cite[Expos\'e V, VI]{SGA5}}, {\cite[1.1]{deligne_lcwii}} {\cite[10.1]{fu_ect}}]\label{definition:artin_rees_mittag_leffler_condition_on_an_inverse_system_of_sheaves_of_abelian_groups_on_a_scheme_for_the_small_etale_site}

Let $X$ be a scheme. Let 
$$\calF = (\calF_n, u_n)_{n \in \bbZ}, \quad u_n: \calF_n \to \calF_{n-1}$$
be an \hyperrefIfExists{definition:system_in_a_category_indexed_by_a_directed_poset}{inverse system}\CrefIfExists{definition:system_in_a_category_indexed_by_a_directed_poset} of sheaves of abelian groups on the \hyperrefIfExists{definition:small_etale_site_of_a_scheme}{(small) étale site $X_{\text{ét}}$}. 
\begin{enumerate}
    \item We say that the inverse system satisfies the \hldef{Mittag–Leffler (ML) condition} if for any integer $n$, there exists an integer $r \geq n$ such that 
    $$\operatorname{im}(\calF_r \to \calF_n) = \operatorname{im}(\calF_t \to \calF_n)$$
    for all $t \geq r$. 

    \TODO{Define $\calF[r]$}
    \item We say that the inverse system satisfies the \hldef{Artin-Rees-Mittag-Leffler (ARML) condition} if for any integer $n$, there exists an integer $r \geq n$ such that 
    $$\operatorname{im}(\calF[r] \to \calF) = \operatorname{im}(\calF[t] \to \calF)$$
    for all $t \geq r$. 

    \item We say that the inverse system is a \hldef{null system} if  there exists an integer $r \geq 0$ such that $\calF[r] \to \calF$ is $0$

\end{enumerate}
