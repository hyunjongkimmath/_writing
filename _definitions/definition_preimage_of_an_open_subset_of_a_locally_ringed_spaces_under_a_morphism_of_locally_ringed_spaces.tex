
\begin{definition}\label{definition:preimage_of_an_open_subset_of_a_locally_ringed_spaces_under_a_morphism_of_locally_ringed_spaces}
Let $(X, \mathcal{O}_X)$ and $(Y, \mathcal{O}_Y)$ be \CrefAndHyperrefIfExist{definition:locally_ringed_space_on_a_topological_space}{locally ringed spaces}, and let 
$$f : (X, \mathcal{O}_X) \to (Y, \mathcal{O}_Y)$$ 
be a \CrefAndHyperrefIfExist{definition:morphism_of_locally_ringed_spaces}{morphism of locally ringed spaces}. The \hldef{preimage of an open subset $V \subseteq Y$} is defined as the \CrefAndHyperrefIfExist{definition:preimage_of_an_open_subset_of_a_ringed_spaces_under_a_morphism_of_ringed_spaces}{preimage of $V$ under $f$} as a \CrefAndHyperrefIfExist{definition:morphism_of_ringed_spaces}{morphism of ringed spaces}:
$$\hlin{f^{-1}(V) := \{ x \in X \mid f(x) \in V \} \subseteq X.}$$

\end{definition}
