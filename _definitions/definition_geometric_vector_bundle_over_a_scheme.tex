\begin{definition} \label{definition:geometric_vector_bundle_over_a_scheme}
Let $S$ be a scheme.
\begin{itemize}
    \item A \hldef{geometric vector bundle of rank $n$ on $S$} is an $S$-scheme $V \to S$ isomorphic to $\mathbf{Spec}(\operatorname{Sym}^\bullet(\mathcal{E}^\vee))$, where $\mathcal{E}$ is a locally free sheaf of rank $n$ on $S$ and $\mathcal{E}^\vee$ is its dual.
    \item A \hldef{geometric line bundle on $S$} is a geometric vector bundle of rank 1. This corresponds to an invertible sheaf $\mathcal{L}$ on $S$, and the total space is $\mathbf{Spec}(\operatorname{Sym}^\bullet(\mathcal{L}^\vee))$.
\end{itemize}
\end{definition}
