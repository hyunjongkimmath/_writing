
\begin{definition} \label{definition:complexification_of_a_free_abelian_group_of_finite_rank}
Let $L$ be a free abelian group of finite rank, and let $k \in \mathbb{Z}$ be an integer. Its \hldef{complexification} is 
$$\hlin{L_{\mathbb{C}} := L \otimes_{\mathbb{Z}} \mathbb{C}.}$$
\CrefIfExists{definition:tensor_product_of_bimodules_of_rings} There is then an abelian group automorphism
$$\hlin{\overline{\cdot}: L_{\mathbb{C}} \to L_{\mathbb{C}}}$$
called \hldef{complex conjugation on $L_{\mathbb{C}}$}.
\end{definition}
