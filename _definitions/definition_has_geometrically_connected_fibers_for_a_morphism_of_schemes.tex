\begin{definition} \label{definition:has_geometrically_connected_fibers_for_a_morphism_of_schemes}
Let $f: X \to Y$ be a \CrefAndHyperrefIfExist{definition:morphism_of_schemes}{morphism of schemes}. 
We say that \hl{$f$ has geometrically connected fibers} if for every \CrefAndHyperrefIfExist{definition:geometric_point_of_a_scheme}{geometric point} $\bar{y}$ of $Y$, the \CrefAndHyperrefIfExist{definition:geometric_fiber_of_a_morphism_of_schemes_at_a_geometric_point}{geometric fiber} $X_{\bar{y}}$ is a \CrefAndHyperrefIfExist{definition:connected_topological_space}{connected topological space}.
\end{definition}
