\begin{definition} \label{definition:normal_and_factorial_scheme}
Let $X$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme}.
\begin{itemize}
    \item The scheme $X$ is called \hldef{normal} if for every point $x \in X$, the local ring $\mathcal{O}_{X,x}$ is an \CrefAndHyperrefIfExist{definition:integral_element_over_a_ring}{integrally closed} \CrefAndHyperrefIfExist{definition:zero_divisor_of_a_ring}{domain}. If $X$ is \CrefAndHyperrefIfExist{definition:integral_scheme}{integral}, this is equivalent to saying that for every affine open subset $U = \operatorname{Spec} A$, the ring $A$ is an integrally closed domain.
    \item The scheme $X$ is called \hldef{factorial} if for every point $x \in X$, the local ring $\mathcal{O}_{X,x}$ is a \CrefAndHyperrefIfExist{definition:unique_factorization_domain}{unique factorization domain} (UFD).
\end{itemize}
\end{definition}
