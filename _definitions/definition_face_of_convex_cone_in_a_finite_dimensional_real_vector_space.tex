\begin{definition}[Face of a Convex Cone] \label{definition:face_of_convex_cone_in_a_finite_dimensional_real_vector_space}
Let $V$ be a \CrefAndHyperrefIfExist{definition:basis_of_a_vector_space_over_a_field}{finite-dimensional} real \CrefAndHyperrefIfExist{definition:vector_space_over_a_field}{vector space}.
Let $\sigma \subset V$ be a \CrefAndHyperrefIfExist{definition:convex_cone_in_a_finite_dimensional_real_vector_space}{convex cone}. 
A \hldef{face of $\sigma$} is a subset
\[
\tau \;=\; \sigma \cap H,
\]
where $H \subset V$ is a \CrefAndHyperrefIfExist{definition:supporting_hyperplane_of_a_convex_cone_in_a_finite_dimensional_real_vector_space}{supporting hyperplane} of $\sigma$.
\end{definition}
