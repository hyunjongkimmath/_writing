\begin{definition}[Flat algebra over a ring] \label{definition:flat_algebra_over_a_ring}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{(not necessarily commutative) ring}. An \CrefAndHyperrefIfExist{definition:algebra_of_a_ring}{$R$-algebra $A$} is said to be \hldef{flat over $R$}, and the \CrefAndHyperrefIfExist{definition:ring_homomorphism}{ring homomorphism} $R \to A$ is called \hldef{flat}, if the underlying $R$-module structure of $A$ is a \CrefAndHyperrefIfExist{definition:flat_left_right_module_over_a_ring}{flat $R$-module}.
\end{definition}
