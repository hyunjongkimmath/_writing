\begin{definition} \label{definition:resolution_of_singularities_of_a_reduced_noetherian_scheme_scheme}
Let $X$ be a \CrefAndHyperrefIfExist{definition:reduced_scheme}{reduced} \CrefAndHyperrefIfExist{definition:resolution_of_singularities_of_a_reduced_noetherian_scheme_scheme}{Noetherian scheme}. A \hldef{resolution of singularities of $X$} is a morphism $\pi: Y \to X$ such that:
\begin{enumerate}
    \item The morphism $\pi$ is \CrefAndHyperrefIfExist{definition:proper_morphism_of_schemes}{proper}.
    \item The scheme $Y$ is \CrefAndHyperrefIfExist{definition:regular_at_a_point_for_a_scheme_regular_scheme_nonsingular_scheme}{regular}.
    \item The morphism $\pi$ restricts to an isomorphism over a dense open subset $U$ of $X$. Typically, $U$ is taken to be the regular locus $X_{\operatorname{reg}}$ of $X$.
\end{enumerate}
\end{definition}
