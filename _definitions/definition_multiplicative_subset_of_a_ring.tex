
\begin{definition} \label{definition:multiplicative_subset_of_a_ring}
Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{not necessarily commutative ring}. A subset $S \subseteq R$ is called a \hldef{multiplicative subset} if 
\begin{itemize}
    \item $1 \in S$ (assuming $R$ has unity),
    \item For all $s, t \in S$, one has $st \in S$.
\end{itemize}
\end{definition}
