\begin{definition}[Automorphism Group of a Field Extension] \label{definition:automorphism_group_of_a_field_extension}
Let $L/K$ be a \CrefAndHyperrefIfExist{definition:extension_of_a_field}{field extension}.  
\begin{itemize}
    \item An \CrefAndHyperrefIfExist{definition:field_automorphism}{automorphism} $\sigma: L \to L$ is called a \hldef{$K$-automorphism} if $\sigma$ is a field automorphism of $L$ that fixes every element of $K$, i.e. $\sigma(a) = a$ for all $a \in K$.
    \item The set of all such $K$-automorphisms of $L$ forms a \CrefAndHyperrefIfExist{definition:group}{group} under composition, called the \hldef{automorphism group of the extension $L/K$}. It is usually denoted by \hl{$\mathrm{Aut}(L/K)$} or \hl{$\Aut_K(L)$}.
\end{itemize}
\end{definition}
