\begin{definition}[Holomorphic and anti-holomorphic tangent bundles of a complex manifold] \label{definition_holomorphic_and_anti_holomorphic_tangent_bundles_of_a_complex_manifold} 
    \TODO{TODO: make a reference to how complex manifolds are equivalent to smooth manifolds with an integrable almost complex structure}
    Let $X$ be a complex manifold of complex dimension $n$ (equivalently a smooth manifold equipped with an integrable almost complex structure $J: TX \to TX$ on the \hyperrefIfExists{definition:tangent_bundle_of_a_smooth_manifold}{tangent bundle} satisfying $J^2 = -\mathrm{id}$\CrefIfExists{theorem:complex_manifold_is_equivalent_to_a_smooth_manifold_of_even_real_dimension_with_an_integrable_almost_complex_structure}). 

    The endomorphism $J$ extends $\mathbb{C}$-linearly to the \hyperrefIfExists{definition:complexification_of_a_real_vector_bundle_over_a_complex_manifold}{complexified} tangent bundle $T_{\mathbb{C}} X$, yielding a decomposition into eigenbundles corresponding to eigenvalues $i$ and $-i$:
    $$T_{\mathbb{C}} X = T^{1,0} X \oplus T^{0,1} X,$$
    where
    \begin{itemize}
        \item \hl{$T^{1,0} X := \ker (J - i \,\mathrm{id})$} is the \hldef{holomorphic tangent bundle}, locally generated by complex-linear tangent vectors,
        \item \hl{$T^{0,1} X := \ker (J + i \,\mathrm{id})$} is the \hldef{anti-holomorphic tangent bundle}, locally generated by conjugate-linear tangent vectors.
    \end{itemize}

    Dually, the complexified cotangent bundle decomposes as
    $$T_{\mathbb{C}}^* X = T^{1,0*} X \oplus T^{0,1*} X,$$
    where 
    \begin{itemize}
        \item \hl{$T^{1,0*} X$} is the dual bundle of $T^{1,0} X$, called the \hldef{holomorphic cotangent bundle},
        \item \hl{$T^{0,1*} X$} is the dual bundle of $T^{0,1} X$, called the \hldef{anti-holomorphic cotangent bundle}.
    \end{itemize}
\end{definition}
