\begin{definition} \label{definition:fan_of_affine_space}
    Let $N = \mathbb{Z}^n$ with basis $\{e_1, \dots, e_n\}$. Let $e_0 = -(e_1 + \dots + e_n)$. The \hldef{fan of affine space $\mathbb{A}^n$} is the \CrefAndHyperrefIfExist{definition:fan_in_a_finite_dimensional_real_vector_space}{fan} \hl{$\Sigma_{\mathbb{A}^n}$} consists of \CrefAndHyperrefIfExist{definition:polyhedral_cone_in_a_finite_dimensional_real_vector_space}{cones} \CrefAndHyperrefIfExist{definition:cone_generated_by_a_subset_of_a_real_vector_space}{generated by} $\{e_1, \dots, e_n\}$ and all of its \CrefAndHyperrefIfExist{definition:face_of_convex_cone_in_a_finite_dimensional_real_vector_space}{faces}.
\end{definition}
