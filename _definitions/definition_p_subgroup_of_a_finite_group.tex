\begin{definition} \label{definition:p_subgroup_of_a_finite_group}
Let $G$ be a \CrefAndHyperrefIfExist{definition:countable_finite_uncountable_sets}{finite} \CrefAndHyperrefIfExist{definition:group}{group} and $p$ be a prime number.
A \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup} $H$ of $G$ is called a \hldef{$p$-subgroup} if the order of $H$ is a power of $p$. 
\end{definition}
