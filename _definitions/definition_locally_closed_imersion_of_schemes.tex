\begin{definition} \label{definition:locally_closed_imersion_of_schemes}
    \begin{enumerate}
        \item Let $f: X \to Y$ be a \CrefAndHyperrefIfExist{definition:morphism_of_schemes}{morphism of schemes}. The morphism $f$ is called a \hldef{locally closed immersion} (or simply an \hldef{immersion}) if it can be factored as the composition $f = j \circ i$, where $i: X \to Z$ is a \CrefAndHyperrefIfExist{definition:open_and_closed_immersions_of_schemes}{closed immersion} into a scheme $Z$, and $j: Z \to Y$ is an \CrefAndHyperrefIfExist{definition:open_and_closed_immersions_of_schemes}{open immersion}. Equivalently, $f$ induces an isomorphism of $X$ onto a locally closed subscheme of $Y$.

        \item Let $X$ be a scheme. A \hldef{locally closed subscheme of $X$} is a scheme $Z$ equipped with a morphism $i: Z \to X$ that is a locally closed immersion.

    \end{enumerate}

\end{definition}
