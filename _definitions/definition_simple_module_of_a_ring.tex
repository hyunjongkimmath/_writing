
\begin{definition} \label{definition:simple_module_of_a_ring}
Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{not necessarily commutative ring}. An $R$-module $M$ is called \hldef{simple} if $M \neq 0$ and the only \CrefAndHyperrefIfExist{definition:submodule_of_a_module_over_a_ring}{submodules of $M$} are $\{0\}$ and $M$ itself.
\end{definition}
