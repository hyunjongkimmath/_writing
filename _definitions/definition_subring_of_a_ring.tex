
\begin{definition}[Subring] \label{definition:subring_of_a_ring}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{not necessarily commutative ring}. A subset $S \subseteq R$ is called a \hldef{subring of $R$} if:
    \begin{itemize}
        \item $S$ is itself a ring with the same operations as $R$ (addition and multiplication inherited from $R$),
        \item the multiplicative identity $1_R$ of $R$ is contained in $S$,
        \item and the additive identity $0_R$ of $R$ is contained in $S$.
    \end{itemize}
    Equivalently, $S \subseteq R$ is a subring if $S$ is nonempty, closed under addition, multiplication, and additive inverses, and contains $1_R$.
\end{definition}
