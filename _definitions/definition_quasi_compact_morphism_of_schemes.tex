\begin{definition}[Quasi-compact morphism of schemes] \label{definition:quasi_compact_morphism_of_schemes}
    Let $f : X \to Y$ be a \CrefAndHyperrefIfExist{definition:morphism_of_schemes}{morphism of schemes}.

    The morphism $f$ is \hldef{quasi-compact} if for every \CrefAndHyperrefIfExist{definition:affine_open_subscheme_of_a_scheme}{open affine subset} \CrefAndHyperrefIfExist{definition:affine_scheme}{$V = \mathrm{Spec}(B) \subseteq Y$}, the \CrefAndHyperrefIfExist{definition:preimage_of_an_open_subset_of_a_scheme_under_a_morphism_of_schemes}{preimage $f^{-1}(V)$} can be covered by finitely many affine open subsets of $X$.

    Equivalently, $f$ is quasi-compact if the inverse image of every \CrefAndHyperrefIfExist{definition:quasi_compact_scheme}{quasi-compact} open subset of $Y$ is quasi-compact in $X$.
\end{definition}
