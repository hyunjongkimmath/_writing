\begin{definition} \label{definition:metric_induced_by_an_absolute_value_on_a_field}
    Let $K$ be a field with an absolute value $|\cdot|$. By the metric induced by $|\cdot|$ we mean the \hyperrefIfExists{definition:extended_metric_induced_by_an_extended_norm_on_a_vector_space_over_a_field_with_absolute_value}{metric induced by $|\cdot|$} regarded as a \hyperrefIfExists{definition:extended_norm_on_a_vector_space_over_a_field_with_absolute_value}{norm}\CrefIfExists{definition:extended_norm_on_a_vector_space_over_a_field_with_absolute_value} on the $1$-dimensional $K$-vector space $K$ with absolute value $|\cdot|_v$. 
\end{definition}
