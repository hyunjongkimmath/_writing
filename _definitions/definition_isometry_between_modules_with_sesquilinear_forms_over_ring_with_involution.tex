\begin{definition} \label{definition:isometry_between_modules_with_sesquilinear_forms_over_ring_with_involution}
    Let $(A, \sigma)$ be a \CrefAndHyperrefIfExist{definition:involution_on_a_ring}{ring with involution}. Let $(M_1,b_1)$ and $(M_2,b_2)$ be \CrefAndHyperref{definition:module_of_a_ring}{(right) modules} equipped with \CrefAndHyperref{definition:sesquilinear_form_on_a_module_over_a_ring_with_involution}{sesquilinear forms}. 

    An \hldef{isometry} $\varphi: (M_1,b_1) \to (M_2,b_2)$ is a \CrefAndHyperref{definition:morphism_between_modules_with_sesquilinear_forms_over_an_involution_ring}{morphism of sesquilinear modules} that is an isomorphism. If an isometry exists between the two sesquilinear modules, then the two forms are said to be \hldef{isometric}.
\end{definition}
