\begin{definition} \label{definition:fan_of_a_hirzebruch_surface}
    Let $N = \mathbb{Z}^2$ with basis $\{e_1, e_2\}$. The \hldef{fan of a Hirzebruch surface $\mathcal{H}_r$} (for $r \geq 0$): Let $N = \mathbb{Z}^2$ with basis $\{e_1, e_2\}$. The fan $\Sigma_{\mathcal{H}_r}$ consists of four maximal \CrefAndHyperrefIfExist{definition:cone_generated_by_a_subset_of_a_real_vector_space}{cones generated by} $\{e_1, e_2\}$, $\{e_2, -e_1\}$, $\{-e_1, -e_2 + r e_1\}$, and $\{-e_2 + r e_1, e_1\}$, along with their \CrefAndHyperrefIfExist{definition:face_of_convex_cone_in_a_finite_dimensional_real_vector_space}{faces}.
\end{definition}
