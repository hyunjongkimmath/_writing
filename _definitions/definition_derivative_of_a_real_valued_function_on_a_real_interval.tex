\begin{definition}[Derivative of a real valued function on a real interval] \label{definition:derivative_of_a_real_valued_function_on_a_real_interval}
    \TODO{topology on $\bbR$}
    \TODO{metric space structure on $\bbR$}
    Let $I \subseteq \mathbb{R}$ be an interval and let $f : I \to \mathbb{R}$ be a function.  Let $a \in I$ be a point such that there is an \CrefAndHyperrefIfExist{definition:neighborhood_and_neighborhood_basis_of_a_point_in_a_topological_space}{open neighborhood} of $a$ contained in $I$.

    The function $f$ is said to be \hldef{differentiable at $a$} if the \CrefAndHyperrefIfExist{definition:limit_of_a_function_between_extended_metric_spaces_at_an_accumulation_point}{limit}
    \[
    \lim_{h\to 0} \frac{f(a+h) - f(a)}{h}
    \]
    exists in $\mathbb{R}$.
    In that case the limit is called the \hldef{derivative of $f$ at $a$} and is denoted \hl{$f'(a)$}.
    If $f : \Omega \to \mathbb{C}$ is a function defined on an \CrefAndHyperrefIfExist{definition:interval_in_the_real_line}{open interval} $\Omega \subseteq \mathbb{R}$ with values in $\mathbb{C}$, the same formula (with values in $\mathbb{C}$) defines differentiability and the derivative $f'(a) \in \mathbb{C}$.
\end{definition}
