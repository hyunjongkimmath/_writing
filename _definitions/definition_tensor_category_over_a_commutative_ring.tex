\begin{definition} \label{definition:tensor_category_over_a_commutative_ring}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative ring}. 
    A \hldef{tensor category over $R$} (or \hldef{$R$-linear symmetric monoidal category}) is an \CrefAndHyperrefIfExist{definition:k_linear_category_over_a_commutative_ring}{$R$-linear category} $\mathcal{C}$ equipped with a \CrefAndHyperrefIfExist{definition:symmetric_monoidal_category}{symmetric monoidal structure} $(\otimes, \mathds{1})$ such that the tensor product functor
    \[
        \otimes: \mathcal{C} \times \mathcal{C} \to \mathcal{C}
    \]
    is $R$-bilinear on morphisms. That is, for all objects $X, X', Y, Y' \in \operatorname{Ob}(\mathcal{C})$, the map
    \[
        \operatorname{Hom}_{\mathcal{C}}(X, Y) \times \operatorname{Hom}_{\mathcal{C}}(X', Y') \to \operatorname{Hom}_{\mathcal{C}}(X \otimes X', Y \otimes Y')
    \]
    given by $(f, g) \mapsto f \otimes g$ is \CrefAndHyperrefIfExist{definition:multilinear_map_of_modules_over_rings}{$R$-bilinear}.
\end{definition}
