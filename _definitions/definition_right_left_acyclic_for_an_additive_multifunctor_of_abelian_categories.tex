\begin{definition}[{\cite[Definition 1.2.3]{SGA4_III}}] \label{definition:right_left_acyclic_for_an_additive_multifunctor_of_abelian_categories}
Let $F: \prod_{i=1}^n \mathscr{A}_i \to \mathscr{B}$ be an \CrefAndHyperrefIfExist{definition:n_ary_additive_functor_between_additive_categories}{additive multifunctor} of \CrefAndHyperrefIfExist{definition:abelian_category}{abelian categories}.
\begin{enumerate}
    \item An object $(A_1, \dots, A_n)$ with $A_i \in \text{ob}(\mathscr{A}_i)$ is \hldef{right acyclic for $F$} if the complex concentrated in degree 0 associated to $(A_i)$ is \CrefAndHyperrefIfExist{definition:defined_at_objects_derivable_and_adapted_objects_for_the_derived_functors_of_an_exact_multifunctor_between_triangulated_categories}{adapted for} $RF$ .
    \item Similarly, $(A_i)$ is \hldef{left acyclic for $F$} if the complex concentrated in degree 0 is \CrefAndHyperrefIfExist{definition:defined_at_objects_derivable_and_adapted_objects_for_the_derived_functors_of_an_exact_multifunctor_between_triangulated_categories}{adapted for} $LF$.
\end{enumerate}

In the case that $n = 1$ and $A_1 \in \calA_1$ has a \CrefAndHyperrefIfExist{definition:injective_and_projective_objects_in_a_category}{projective} \CrefAndHyperrefIfExist{definition:left_right_resolution_of_a_class_of_objects_in_an_abelian_category}{resolution} (resp. injective resolution), the notion of left acyclic for $F$ (resp. right acyclic for $F$) is equivalent to the notion of \CrefAndHyperref{definition:F_acyclic_object_for_a_left_or_right_functor_between_abelian_categories}{$F$-acyclicity}.
\end{definition}
