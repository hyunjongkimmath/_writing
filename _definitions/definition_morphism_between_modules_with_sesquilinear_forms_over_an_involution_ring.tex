\begin{definition} \label{definition:morphism_between_modules_with_sesquilinear_forms_over_an_involution_ring}
    Let $(A, \sigma)$ be a \CrefAndHyperrefIfExist{definition:involution_on_a_ring}{ring with involution}. Let $(M_1,b_1)$ and $(M_2,b_2)$ be \CrefAndHyperref{definition:module_of_a_ring}{(right) modules} equipped with \CrefAndHyperref{definition:sesquilinear_form_on_a_module_over_a_ring_with_involution}{sesquilinear forms}. 
    A \hldef{morphism} $(M_1,b_1) \to (M_2,b_2)$ between the sesquilinear modules is a \CrefAndHyperref{definition:homomorphism_of_modules_over_a_ring}{(right) $A$-module homomorphism} such that $\Sesq(\varphi)(b_2) = b_1$, i.e. 
    $$b_2(\varphi(x), \varphi(y)) = b_1(x,y)$$
    for all $x,y \in M_1$.
\end{definition}
