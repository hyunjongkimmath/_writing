
\begin{definition} \label{definition:localization_of_a_commutative_ring_by_a_multiplicative_subset}
Let $R$ be a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative ring with unity} and let $S \subseteq R$ be a \CrefAndHyperrefIfExist{definition:multiplicative_subset_of_a_ring}{multiplicative subset}.  
The \hldef{localization of $R$ at $S$}, denoted by \hl{$S^{-1}R$}, is the ring whose elements are equivalence classes of pairs $(r,s) \in R \times S$ under the equivalence relation
$$(r,s) \sim (r',s') \iff \exists u \in S \ \text{such that}\ u(sr' - s'r) = 0.$$
Write \hl{$\frac{r}{s}$} for the equivalence class of $(r,s)$. Addition and multiplication on representatives are defined by
\begin{align*}
\frac{r}{s} + \frac{r'}{s'} &= \frac{rs' + r's}{ss'}, \\
\frac{r}{s} \cdot \frac{r'}{s'} &= \frac{rr'}{ss'}.
\end{align*}
The map $r \mapsto \frac{r}{1}$ defines a \CrefAndHyperrefIfExist{definition:ring_homomorphism}{ring homomorphism}; therefore, $S^{-1}R$ is naturally an \CrefAndHyperrefIfExist{definition:algebra_of_a_ring}{$R$-algebra}.

Given an element $f \in R$, the localization of $R$ at $S = \{f^n: n\geq 0 \}$ is denoted by \hl{$R_f$}.

If $P$ is a \CrefAndHyperrefIfExist{definition:prime_and_maximal_ideal_of_a_ring}{prime ideal of $R$}, then \hldef{$R_P := S^{-1}R$} with $S = R \setminus P$ is called the \hldef{localization of $R$ at $P$}. It is a \CrefAndHyperrefIfExist{definition:local_ring}{local ring} whose \CrefAndHyperrefIfExist{definition:prime_and_maximal_ideal_of_a_ring}{maximal ideal} is given by 
$$S^{-1}P = \{ \frac{p}{s} \in R_P: p \in P \}.$$
\end{definition}
