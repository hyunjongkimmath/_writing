\begin{definition}[Holomorphism and biholomorphism of complex manifolds] \label{definition:holomorphism_and_biholomoprhism_of_complex_manifolds}
    \TODO{}
Let $M$ and $N$ be \CrefAndHyperrefIfExist{definition:complex_analytic_manifold}{complex manifolds}.

A continuous map $f : M \to N$ is called a \hldef{holomorphism of complex manifolds} if for every chart $(U,\varphi)$ on $M$ and every chart $(V,\psi)$ on $N$ with $f(U) \subseteq V$, the coordinate expression
\[
\psi \circ f \circ \varphi^{-1} : \varphi(U) \to \psi(V)
\]
is a holomorphic map between open subsets of complex Euclidean spaces.
A \hldef{biholomorphism of complex manifolds} is a bijective holomorphism $f : M \to N$ whose inverse map $f^{-1} : N \to M$ is also a holomorphism; equivalently, $f$ is a homeomorphism and both $f$ and $f^{-1}$ are holomorphic in the above sense.
\end{definition}
