\begin{definition} \label{definition:ind_etale_morphism_of_schemes}
    \TODO{(affine) open cover}
Let $X$ be a scheme. A morphism of schemes $f \colon Y \to X$ is called \hldef{ind-étale} if for every \CrefAndHyperrefIfExist{definition:affine_open_subscheme_of_a_scheme}{affine open} $U = \operatorname{Spec} R \subseteq X$, the \CrefAndHyperrefIfExist{definition:preimage_of_an_open_subset_of_a_scheme_under_a_morphism_of_schemes}{inverse image} $f^{-1}(U)$ admits an affine open cover $\{V_j = \operatorname{Spec} B_j\}$ such that each $R$-algebra $B_j$ is \CrefAndHyperrefIfExist{definition:ind_etale_algebra_over_a_commutative_ring}{ind-étale over} $R$. Equivalently, $f$ is locally of the form $\underline{\operatorname{Spec}}(\mathcal{A}) \to X$\CrefIfExists{definition:relative_spectrum_of_a_quasi_coherent_sheaf_of_O_algebras_on_a_scheme} where $\mathcal{A}$ is a \CrefAndHyperrefIfExist{definition:quasi_coherent_sheaf_on_a_general_scheme}{quasi-coherent} \CrefAndHyperrefIfExist{definition:algebra_over_a_sheaf_of_rings_on_a_site}{$\mathcal{O}_X$-algebra} that is locally (on affine opens of $X$) a \CrefAndHyperrefIfExist{definition:projective_and_inductive_limits_in_categories}{filtered colimit} of \CrefAndHyperrefIfExist{definition:etale_O_X_algebra_over_a_scheme}{étale $\mathcal{O}_X$-algebras}.
\end{definition}
