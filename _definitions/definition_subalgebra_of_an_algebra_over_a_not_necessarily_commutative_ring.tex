\begin{definition}[Subalgebra of an algebra over a ring] \label{definition:subalgebra_of_an_algebra_over_a_not_necessarily_commutative_ring}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{(not necessarily commutative) ring} and let $A$ be an \CrefAndHyperrefIfExist{definition:algebra_of_a_ring}{$R$-algebra}.

    A subset $B \subseteq A$ is an \hldef{$R$-subalgebra} if:
    \begin{itemize}
        \item $B$ is a \CrefAndHyperrefIfExist{definition:subring_of_a_ring}{subring of $A$},
        \item $B$ is closed under scalar multiplication by $R$, i.e., for all $r \in R$ and $b \in B$, we have $r \cdot b \in B$.
    \end{itemize}

    In particular, $B$ inherits the structure of an $R$-algebra as a subobject of $A$.
\end{definition}
