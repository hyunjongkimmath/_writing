\begin{definition}[Lebesgue measure on $\mathbb{R}^n$] \label{definition:lebesgue_measure_on_Rn}
    \TODO{topology on $\bbR^n$}
    Let $n \ge 1$ be an integer and consider $\mathbb{R}^n$ with its \CrefAndHyperrefIfExist{definition:borel_sigma_algebra_of_a_topological_space}{Borel $\sigma$-algebra $\mathcal{B}(\mathbb{R}^n)$} generated by the open subsets.

    The \hldef{Lebesgue measure on $\mathbb{R}^n$} is the unique \CrefAndHyperrefIfExist{definition:measure_on_a_measurable_space_and_measure_space}{measure} $\lambda^n : \mathcal{B}(\mathbb{R}^n) \to [0,\infty]$ satisfying:
    \begin{enumerate}
    \item for any rectangle $R = \prod_{k=1}^n (a_k,b_k) \subseteq \mathbb{R}^n$ \CrefIfExists{definition:interval_in_the_real_line} with $a_k < b_k$, one has
    \[
    \lambda^n(R) = \prod_{k=1}^n (b_k - a_k),
    \]
    \item $\lambda^n$ is translation invariant: $\lambda^n(A + x) = \lambda^n(A)$ for all $A \in \mathcal{B}(\mathbb{R}^n)$ and $x \in \mathbb{R}^n$,
    \end{enumerate}
    where $A + x = \{a + x : a \in A\}$.
\end{definition}
