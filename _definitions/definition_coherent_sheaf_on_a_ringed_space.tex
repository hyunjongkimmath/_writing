\begin{definition} \label{definition:coherent_sheaf_on_a_ringed_space}
Let $(X, \mathcal{O}_X)$ be a \CrefAndHyperrefIfExist{definition:ringed_space}{ringed space}. An $\mathcal{O}_X$-module sheaf $\mathcal{F}$ is \hldef{coherent} if:
\begin{enumerate}
  \item $\mathcal{F}$ is of finite type, i.e., locally there exists a surjection $\mathcal{O}_U^n \twoheadrightarrow \mathcal{F}|_U$ for some finite $n$, and
  \item the kernel of any morphism $\mathcal{O}_U^m \to \mathcal{F}|_U$ is also of finite type, i.e., $\mathcal{F}$ is locally finitely presented.
\end{enumerate}
Equivalently, $\mathcal{F}$ is locally the cokernel of a morphism between finite free $\mathcal{O}_U$-modules.

\TextIfExists{definition:coherent_sheaf_on_a_ringed_site}{
Equivalently, a coherent sheaf on the ringed space $(X, \calO_X)$ is a \CrefAndHyperrefIfExist{definition:coherent_sheaf_on_a_ringed_site}{coherent sheaf} on $(X, \calO_X)$ \CrefAndHyperrefIfExist{definition:ringed_space}{as a} \CrefAndHyperrefIfExist{definition:ringed_site}{ringed site}. 
}
\end{definition}
