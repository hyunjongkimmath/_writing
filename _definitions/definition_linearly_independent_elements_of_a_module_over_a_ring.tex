\begin{definition} \label{definition:linearly_independent_elements_of_a_module_over_a_ring}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{(not necessarily commutative) ring} and let $M$ be an \CrefAndHyperrefIfExist{definition:module_of_a_ring}{$R$-module}.
    \begin{enumerate}
        \item If $M$ is a \CrefAndHyperrefIfExist{definition:module_of_a_ring}{left $R$-module}, a finite subset $S = \{x_1, \dots, x_n\} \subseteq M$ is said to be \hldef{linearly independent over $R$} if, whenever $r_1, \dots, r_n \in R$ satisfy
        $$r_1 x_1 + r_2 x_2 + \cdots + r_n x_n = 0,$$
        it follows that each $r_i = 0$.

        \item If $M$ is a \CrefAndHyperrefIfExist{definition:module_of_a_ring}{right $R$-module}, a finite subset $S = \{x_1, \dots, x_n\} \subseteq M$ is said to be \hldef{linearly independent over $R$} if, whenever $r_1, \dots, r_n \in R$ satisfy
        $$x_1 r_1 + x_2 r_2 + \cdots + x_n r_n = 0,$$
        it follows that each $r_i = 0$.

        \item If $M$ is a \CrefAndHyperrefIfExist{definition:module_of_a_ring}{two-sided $R$-module}, a finite subset $S = \{x_1, \dots, x_n\} \subseteq M$ is said to be \hldef{linearly independent over $R$} if, whenever $r_1, \dots, r_n, s_1, \dots, s_n \in R$ satisfy
        $$r_1 x_1 s_1 + r_2 x_2 s_2 + \cdots + r_n x_n s_n = 0,$$
        it follows that each $r_i s_i = 0$ in $R$. Equivalently, no nontrivial pair of coefficient actions on the $x_i$ yields the zero element of $M$.
    \end{enumerate}
    An arbitrary subset $S \subseteq M$ is \hldef{linearly independent over $R$} if every finite subset of $S$ is linearly independent.
\end{definition}
