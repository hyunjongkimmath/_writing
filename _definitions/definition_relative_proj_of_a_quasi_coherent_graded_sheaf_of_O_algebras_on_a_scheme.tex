\begin{definition}  \label{definition:relative_proj_of_a_quasi_coherent_graded_sheaf_of_O_algebras_on_a_scheme}
Let $S$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme} and let $\mathcal{S} = \bigoplus_{d \geq 0} \mathcal{S}_d$ be a \CrefAndHyperrefIfExist{definition:quasi_coherent_sheaf_on_a_general_scheme}{quasi-coherent} graded sheaf of \CrefAndHyperrefIfExist{definition:algebra_over_a_sheaf_of_rings_on_a_site}{$\mathcal{O}_S$-algebras} such that $\mathcal{S}_0 = \mathcal{O}_S$ and $\mathcal{S}$ is locally generated by $\mathcal{S}_1$ as an $\mathcal{S}_0$-algebra. The \hldef{relative Proj of $\mathcal{S}$}, denoted by notations such as \hl{$\mathbf{Proj}(\mathcal{S})$}, \hl{$\mathcal{P}roj(\mathcal{S})$}, or \hl{$\underline{\operatorname{Proj}}(\calA)$}, is an \CrefAndHyperrefIfExist{definition:scheme_over_a_scheme}{$S$-scheme} $\pi: \mathbf{Proj}(\mathcal{S}) \to S$ constructed as follows:

\begin{enumerate}
    \item For every \CrefAndHyperrefIfExist{definition:affine_open_subscheme_of_a_scheme}{affine open} subset $U = \operatorname{Spec} A \subseteq S$, let $S(U) = \Gamma(U, \mathcal{S})$ be the graded $A$-algebra of sections over $U$. We define the scheme $X_U$ to be the projective spectrum of this ring:
    $$\hlin{X_U = \operatorname{Proj}(S(U)).}$$
    This comes with a natural structure morphism $\pi_U: X_U \to U$.
    \item For any inclusion of affine open subsets $V \subseteq U$, the restriction map $\mathcal{S}(U) \to \mathcal{S}(V)$ induces a canonical isomorphism of schemes $\pi_U^{-1}(V) \cong X_V$.
    \item The global scheme $\mathbf{Proj}(\mathcal{S})$ is obtained by gluing the family $\{X_U\}$ for all affine open $U \subseteq S$ along these canonical isomorphisms.
\end{enumerate}
The scheme $\mathbf{Proj}(\mathcal{S})$ comes equipped with a canonical \CrefAndHyperrefIfExist{definition:algebraic_vector_bundle_over_a_scheme}{invertible sheaf} $\mathcal{O}_{\mathbf{Proj}(\mathcal{S})}(1)$, which is defined locally on each $X_U = \operatorname{Proj}(S(U))$ as the sheaf associated to the twisted module $S(U)(1)$.
\end{definition}
