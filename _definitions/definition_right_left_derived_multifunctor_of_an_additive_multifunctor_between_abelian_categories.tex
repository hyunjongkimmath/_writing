\begin{definition} \label{definition:right_left_derived_multifunctor_of_an_additive_multifunctor_between_abelian_categories}
Let $F: \prod_{i=1}^n \mathscr{A}_i \to \mathscr{B}$ be an \CrefAndHyperrefIfExist{definition:n_ary_additive_functor_between_additive_categories}{additive multifunctor} of \CrefAndHyperrefIfExist{definition:abelian_category}{abelian categories}.

The \hldef{right (resp. left) derived multifunctors of $F$} are the \CrefAndHyperrefIfExist{definition:right_left_derived_functor_of_a_exact_mixed_variant_multifunctor_between_triangulated_categories_relative_to_saturated_multiplicative_systems}{derived functors} \hl{$RF$} (resp. \hl{$LF$}) of the \CrefAndHyperrefIfExist{theorem:additive_functor_between_abelian_categories_induces_an_exact_functor_between_homotopy_categories_as_triangulated_categories}{induced multifunctors} 
\begin{itemize}
    \item $K(\prod \mathscr{A}_i) \to K(\mathscr{B})$
    \item $K^+(\prod \mathscr{A}_i) \to K^+(\mathscr{B})$
    \item $K^-(\prod \mathscr{A}_i) \to K^-(\mathscr{B})$
\end{itemize}
(Recall \Cref{proposition:homotopy_category_of_an_additive_category_is_a_triangulated_category}) relative to the \CrefAndHyperrefIfExist{definition:saturated_multiplicative_system_in_a_category}{systems} of \CrefAndHyperrefIfExist{definition:quasi_isomorphism_of_chain_complexes_of_objects_in_an_abelian_category}{quasi-isomorphisms} in the respective \CrefAndHyperrefIfExist{definition:homotopy_category_of_chain_complexes_of_an_additive_category}{homotopy categories}. In particular, the functor $RF$ refers to one of the functors
\begin{align*}
    D^+(\prod \mathscr{A}_i) \to \operatorname{Ind} D^+(\mathscr{B})
    D(\prod \mathscr{A}_i) \to \operatorname{Ind} D(\mathscr{B})
\end{align*}
and the functor $LF$ refers to one of the functors
\begin{align*}
    D^-(\prod \mathscr{A}_i) \to \operatorname{Pro} D^-(\mathscr{B})
    D(\prod \mathscr{A}_i) \to \operatorname{Pro} D(\mathscr{B})
\end{align*}
(\Cref{definition:derived_category_of_an_abelian_category}) (\Cref{definition:ind_pro_category_of_a_locally_small_category}).
\end{definition}
