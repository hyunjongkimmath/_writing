\begin{definition} \label{definition:pre_abelian_category}
    A category $\calC$ is \hldef{pre-abelian} if it is \CrefAndHyperrefIfExist{definition:additive_category_preadditive_category}{additive} and every morphism hsa a \CrefAndHyperrefIfExist{definition:kernel_and_cokernel_of_a_morphism_in_a_category}{kernel and a cokernel}.
\end{definition}
