
\begin{definition}[Degree of an Extension, Finite Extension] \label{definition:degree_of_a_field_extension}
Let $L/K$ be a \CrefAndHyperrefIfExist{definition:extension_of_a_field}{field extension}. The \hldef{degree} of the extension \hl{$[L:K] := \dim_K(L)$}, i.e. the dimension of $L$ as a $K$-vector space\CrefIfExists{lemma:extension_of_a_field_is_a_vector_space}. If $[L:K] < \infty$, the extension $L/K$ is called a \hldef{finite field extension} and $L$ is said to be a \hldef{finite extension of $K$}.
\end{definition}
