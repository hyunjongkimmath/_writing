\begin{definition}[Dense subset] \label{definition:dense_subset_of_a_topological_space}
Let $(X, \mathcal{T})$ be a topological space. A subset $D \subseteq X$ is called \hldef{dense in $X$} if the \hyperrefIfExists{closure_of_a_subspace_of_a_topological_space}{closure of $D$}\CrefIfExists{closure_of_a_subspace_of_a_topological_space} equals $X$, i.e.,
\[
\overline{D} = X.
\]
Equivalently, $D$ is dense in $X$ if every non-empty open set $U \in \mathcal{T}$ intersects $D$, that is,
\[
\forall U \in \mathcal{T}, \; U \neq \emptyset \implies U \cap D \neq \emptyset.
\]
\end{definition}
