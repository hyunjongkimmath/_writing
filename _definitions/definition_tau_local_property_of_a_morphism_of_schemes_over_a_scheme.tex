\begin{definition}[{\cite[Tag 02KO]{stacks-project}}] \label{definition:tau_local_property_of_a_morphism_of_schemes_over_a_scheme}
    Let $S$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme}. Let $\mathcal{P}$ be a property of morphisms of schemes over $S$. For $\tau$ some \CrefAndHyperrefIfExist{definition:grothendieck_topology_on_a_category_site_covering_sieve_topologically_generating_family}{Grothendieck topology} on $\textit{Sch}/S$\footnote{for example, $$\tau \in \{\text{fpqc}, \text{fppf}, \text{syntomic}, \text{smooth}, \text{\'etale}, \text{Zariski}\}.$$},  we say that $\mathcal{P}$ is \hldef{$\tau$-local on the base} or \hldef{$\tau$-local on the target}, or \hldef{local on the base for the $\tau$-topology} if for any $\tau$-covering $\{Y_i \to Y\}_{i \in I}$, and any morphism of schemes $f: X \to Y$ over $S$, the morphism $f$ has $\mathcal{P}$ if and only if $Y_i \times_Y X \to Y_i$\CrefIfExists{definition:cartesian_product_of_two_objects_in_a_category_over_an_object} has $\mathcal{P}$ for every $i$. 
\end{definition}
