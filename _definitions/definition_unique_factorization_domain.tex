
\begin{definition}[Unique factorization domain (UFD) / Factorial ring] \label{definition:unique_factorization_domain}
\TODO{define irreducible elements}
An \CrefAndHyperrefIfExist{definition:zero_divisor_of_a_ring}{integral domain} $R$ is called a \hldef{unique factorization domain (UFD)} or \hldef{factorial ring} if every nonzero nonunit element of $R$ can be factored as a product of irreducible elements uniquely up to order and \CrefAndHyperrefIfExist{definition:unit_of_a_ring}{units}.
\end{definition}
