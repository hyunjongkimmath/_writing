\begin{definition} \label{definition:geometric_property_of_a_scheme_over_a_field}
Let $k$ be a field.
A \hldef{geometric property of a $k$-scheme} $X$ (or a \hldef{property "holding geometrically"}) is a property that holds for the base change $X_{\bar{k}} := X \times_k \operatorname{Spec}(\bar{k})$, where $\bar{k}$ is an \CrefAndHyperrefIfExist{definition:algebraically_closure_of_a_field}{algebraic closure} of $k$.
In particular, we may speak of the following properties of $X$:
\begin{enumerate}
    % TOPOLOGICAL PROPERTIES
    \item \CrefAndHyperrefIfExist{definition:geometrically_connected_scheme_over_a_field}{geometrically connected} 
    \item geometrically \CrefAndHyperrefIfExist{definition:irreducible_scheme}{irreducible}
    
    % STRUCTURAL/INTEGRITY PROPERTIES
    \item geometrically \CrefAndHyperrefIfExist{definition:reduced_scheme}{reduced}
    \item geometrically \CrefAndHyperrefIfExist{definition:integral_scheme}{integral} \TODO{clarify that this is equivalent to being both geometrically reduced and geometrically irreducible}
    
    \TODO{go thorugh the below}
    % SINGULARITIES AND LOCAL PROPERTIES
    \item geometrically \CrefAndHyperrefIfExist{definition:normal_scheme}{normal}
    \item geometrically \CrefAndHyperrefIfExist{definition:regular_scheme}{regular} \TODO{Note: for schemes of finite type over a perfect field, this is equivalent to being regular}
    \item geometrically \CrefAndHyperrefIfExist{definition:unibranch_scheme}{unibranch}
    \item geometrically \CrefAndHyperrefIfExist{definition:cohen_macaulay_scheme}{Cohen-Macaulay}
    \item geometrically \CrefAndHyperrefIfExist{definition:locally_factorial_scheme}{locally factorial} (or geometrically parafactorial)
    
    % SERRE CONDITIONS
    \item geometrically satisfying \CrefAndHyperrefIfExist{definition:serre_condition_sn}{Serre's condition $S_n$}
    \item geometrically satisfying \CrefAndHyperrefIfExist{definition:serre_condition_rn}{Serre's condition $R_n$}
\end{enumerate}

\end{definition}
