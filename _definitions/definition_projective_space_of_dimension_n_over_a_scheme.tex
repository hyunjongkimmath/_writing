\begin{definition} \label{definition:projective_space_of_dimension_n_over_a_scheme}
Let $S$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme} and let $n \geq 0$ be an integer. The \hldef{projective space of dimension $n$ over $S$}, denoted by \hl{$\mathbb{P}^n_S$}, is constructed as follows:
\begin{enumerate}
    \item If $S = \operatorname{Spec} A$ is an \CrefAndHyperrefIfExist{definition:affine_scheme}{affine scheme}, then $\mathbb{P}^n_S$ is the ``usual'' projective space over the ring $A$, defined as the \CrefAndHyperrefIfExist{definition:projective_spectrum_of_a_Z_geq_0_graded_commutative_ring}{projective spectrum} of the polynomial ring in $n+1$ variables with coefficients in $A$:
    $$\mathbb{P}^n_{\operatorname{Spec} A} = \operatorname{Proj}(A[T_0, \dots, T_n]);$$
    here, each variable $T_i$ is graded in degree $1$, and elements of $A$ are graded in degree $0$.
    \item For a general scheme $S$, let $\{U_i = \operatorname{Spec} A_i\}_{i \in I}$ be an affine open covering of $S$. For each $i$, let $X_i = \mathbb{P}^n_{U_i} = \operatorname{Proj}(A_i[T_0, \dots, T_n])$. For any open immersion $U_{ij} = U_i \cap U_j \hookrightarrow U_i$, the restriction maps on coefficients induce canonical isomorphisms $X_i|_{U_{ij}} \cong X_j|_{U_{ij}}$. The scheme $\mathbb{P}^n_S$ is obtained by gluing the family $\{X_i\}$ along these isomorphisms.
\end{enumerate}
\TextIfExists{definition:relative_proj_of_a_quasi_coherent_graded_sheaf_of_O_algebras_on_a_scheme}{
Equivalently, $\mathbb{P}^n_S$ is the \CrefAndHyperrefIfExist{definition:relative_proj_of_a_quasi_coherent_graded_sheaf_of_O_algebras_on_a_scheme}{relative Proj} of the graded sheaf of polynomial algebras in $n+1$ variables over $\mathcal{O}_S$:
$$\hlin{\mathbb{P}^n_S = \mathbf{Proj}(\mathcal{O}_S[T_0, \dots, T_n]).}$$
}
\end{definition}
