
\begin{definition}[Centralizer] \label{definition:centralizer_of_a_subgroup_of_a_group}
Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} and let $S \subseteq G$ be a subset. The \hldef{centralizer of $S$ in $G$} is defined as
$$\hlin{C_G(S) = \{ g \in G : g s = s g \text{ for all } s \in S \}}.$$
\end{definition}
