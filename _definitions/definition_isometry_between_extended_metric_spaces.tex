
\begin{definition}[Isometry for extended metric spaces] \label{definition:isometry_between_extended_metric_spaces}
Let $(X, d_X)$ and $(Y, d_Y)$ be \hyperrefIfExists{definition:extended_metric_on_a_set}{extended metric spaces}\CrefIfExists{definition:extended_metric_on_a_set}. A map $f : X \to Y$ is called an \hldef{isometry} if for all $x_1, x_2 \in X$,
\[
d_Y( f(x_1), f(x_2) ) = d_X( x_1, x_2 ).
\]
An isometry is automatically injective. If, in addition, $f$ is surjective, it is called an \hldef{isometric isomorphism}, and $X$ and $Y$ are said to be \hldef{isometric}.
\end{definition}
