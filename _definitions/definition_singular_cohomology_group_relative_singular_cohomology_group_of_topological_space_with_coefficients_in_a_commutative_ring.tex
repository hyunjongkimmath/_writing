\begin{definition} \label{definition:singular_cohomology_group_relative_singular_cohomology_group_of_topological_space_with_coefficients_in_a_commutative_ring}
    Let $X$ be a \CrefAndHyperrefIfExist{definition:topological_space}{topological space} and $R$ a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative ring} with $1$. 
    The \hldef{$n$-th singular cohomology group of $X$ with coefficients in $R$} is
    $$\hlin{H^n(X;R) = \ker(\delta^n)/\operatorname{im}(\delta^{n-1}).}$$
    For a subspace $A \subseteq X$ the \hldef{relative cohomology groups of $(X,A)$} are
    $$\hlin{H^n(X,A;R) = H^n(C^*(X,A;R)).}$$
    In other words, these singular cohomology groups are the \CrefAndHyperrefIfExist{definition:homology_and_cohomology_objects_for_a_chain_complex_in_an_additive_category}{cohomology $R$-modules} of the \CrefAndHyperrefIfExist{definition:singular_cochain_group_and_complex_relative_singular_cochain_group_and_complex_of_topological_space_with_coefficients_in_a_commutative_unital_ring}{singular cochain complex $C^*(X;R)$ and the relative singular cochain complex $C^*(X,A;R)$} respectively.
\end{definition}
