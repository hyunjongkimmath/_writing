
\begin{definition}[Laplacians on a Kähler manifold]  \label{definition_}
    %label{definition_}
Let $(X, \omega)$ be a Kähler manifold. Define \hldef{adjoint Dolbeault operators} \hl{$\partial^*$} and \hl{$\overline{\partial}^*$} by
% the Hermitian adjoints with respect to the Kähler metric and inner product by
$$\hlin{\partial^* = - \star \overline{\partial} \star, \quad \overline{\partial}^* = - \star \partial \star.}$$

Then the \hldef{Dolbeault Laplacians} are given by
$$\hlin{\Delta_\partial := \partial \partial^* + \partial^* \partial, \quad \Delta_{\overline{\partial}} := \overline{\partial} \overline{\partial}^* + \overline{\partial}^* \overline{\partial}.}$$

Define the \hldef{Hodge Laplacian on $(p,q)$}-forms by
$$\hlin{\Delta := d d^* + d^* d,}$$
where \hl{$d^* = \partial^* + \overline{\partial}^*$}.
\end{definition}
