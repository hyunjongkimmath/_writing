\begin{definition} \label{definition:cotangent_space_and_tangent_space_of_a_scheme_at_a_point}
Let $X$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme} and let $x \in X$ be a point. Let \CrefAndHyperrefIfExist{definition:stalk_of_a_presheaf_on_a_topological_space_at_a_point}{$\mathcal{O}_{X,x}$} be the local ring at $x$ with \CrefAndHyperrefIfExist{definition:prime_and_maximal_ideal_of_a_ring}{maximal ideal} $\mathfrak{m}_x$ and \CrefAndHyperrefIfExist{definition:residue_field_of_a_ring_at_a_point}{residue field} $\kappa(x) = \mathcal{O}_{X,x}/\mathfrak{m}_x$.
\begin{itemize}
    \item The \hldef{cotangent space of $X$ at $x$}, denoted by \hl{$T^*_x X$}, is the \CrefAndHyperrefIfExist{definition:zariski_cotangent_space_of_tangent_space_of_a_local_ring}{Zariski cotangent space} of the local ring $\mathcal{O}_{X,x}$, i.e., $T^*_x X = \mathfrak{m}_x/\mathfrak{m}_x^2$.
    \item The \hldef{tangent space of $X$ at $x$}, denoted by \hl{$T_x X$}, is the \CrefAndHyperrefIfExist{definition:zariski_cotangent_space_of_tangent_space_of_a_local_ring}{Zariski tangent space} of the local ring $\mathcal{O}_{X,x}$, given by
    $$T_x X = \operatorname{Hom}_{\kappa(x)}(\mathfrak{m}_x/\mathfrak{m}_x^2, \kappa(x)).$$
\end{itemize}
\end{definition}
