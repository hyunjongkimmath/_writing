\begin{definition}[Harmonic forms on a smooth oriented \hyperrefIfExists{definition:riemannian_metric_of_a_smooth_manifold_and_a_riemannian_manifold}{riemannian manifold}] \label{definition:harmonic_forms_on_a_smooth_oriented_riemannian_manifold}
Let $M$ be a \hyperrefIfExists{definition:C_k_manifold}{smooth}, oriented \hyperrefIfExists{definition:riemannian_metric_of_a_smooth_manifold_and_a_riemannian_manifold}{Riemannian manifold} of dimension $n$ with Riemannian metric $g$.
A differential form $\omega \in \Omega^k(M)$ is called \hldef{harmonic} if it satisfies
$$
\Delta \omega = 0.
$$
Equivalently, by standard Hodge theory,
$$
\omega \quad \text{harmonic} \quad\iff\quad d\omega = 0 \quad\text{and}\quad \delta \omega = 0.
$$
Denote the space of harmonic $k$-forms by
$$\hlin{\mathcal{H}^k(M) := \{ \omega \in \Omega^k(M) : \Delta \omega = 0 \}.}$$
\end{definition}
