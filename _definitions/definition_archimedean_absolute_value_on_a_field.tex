\begin{definition} \label{definition:archimedean_absolute_value_on_a_field}
    Let $F$ be a \CrefAndHyperrefIfExist{definition:field}{field}. An \hyperrefIfExists{definition:absolute_value_on_a_field}{absolute value on $F$} is said to be \hldef{non-archimedean} if 
    $$|x+y| \leq \max(|x|, |y|) \quad \text{for all } x,y \in F$$
    and is said to be \hldef{archimedean} otherwise.
\end{definition}
