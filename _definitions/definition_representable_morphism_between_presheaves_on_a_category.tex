\begin{definition} \label{definition:representable_morphism_between_presheaves_on_a_category}
    \TODO{make this definition be for more general presheaves valued in more general caetgories}
    Let $\mathcal{C}$ be a category and let $F,G$ be \CrefAndHyperrefIfExist{definition:presheaf_on_a_category}{presheaves} on $\mathcal{C}$ valued in sets, i.e. functors $\mathcal{C}^{\op} \to \Sets$. We say that a morphism $F \to G$ is \hldef{representable} if for any object $T$ of $\mathcal{C}$ and morphism $h_T \to G$ of presheaves, the fibered product $h_T \times_G F$ is \CrefAndHyperrefIfExist{definition:representable_functor_on_a_category_enriched_in_a_monoidal_category}{representable}.
\end{definition}
