\begin{definition} \label{definition:cone_generated_by_a_subset_of_a_real_vector_space}
Let $V$ be a real \CrefAndHyperrefIfExist{definition:vector_space_over_a_field}{vector space} and let $S \subset V$ be a subset. The \hldef{cone generated by $S$}, denoted by \hl{$\operatorname{cone}(S)$}, is the set of all non-negative linear combinations of elements of $S$:
$$\hlin{\operatorname{cone}(S) = \left\{ \sum_{v \in S} \lambda_v v \mid \lambda_v \ge 0, \text{ and } \lambda_v = 0 \text{ for all but finitely many } v \right\}.}$$
% If $S = \{v_1, \dots, v_k\}$ is a finite set, the cone generated by $S$ is called a \hldef{polyhedral cone}.
\end{definition}
