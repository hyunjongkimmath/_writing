\begin{definition} \label{definition:prime_subfield_of_a_field}
    \TODO{finite field, rational numbers}
Let $K$ be a field. The \hldef{prime subfield of $K$} is the intersection of all \CrefAndHyperrefIfExist{definition:extension_of_a_field}{subfields} of $K$.
\begin{itemize}
    \item If $\operatorname{char}(K) = p > 0$, the prime subfield is isomorphic to the finite field $\mathbb{F}_p \cong \mathbb{Z}/p\mathbb{Z}$.
    \item If $\operatorname{char}(K) = 0$, the prime subfield is isomorphic to the field of rational numbers $\mathbb{Q}$.
\end{itemize}
\end{definition}
