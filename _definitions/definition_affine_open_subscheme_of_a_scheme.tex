
\begin{definition} \label{definition:affine_open_subscheme_of_a_scheme}
Let $X$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme}. A subset $U \subseteq X$ is called an \hldef{affine open subset/subscheme of $X$} if $U$ is open in the underlying topological space of $X$ and the restricted scheme $(U, \mathcal{O}_X|_U)$ is isomorphic, as a scheme, to \CrefAndHyperrefIfExist{definition:affine_scheme}{$\operatorname{Spec} A$} for some commutative ring $A$.
\end{definition}
