\begin{definition} \label{definition:automorphism_of_a_group_automorphism_group_of_a_group}
Let $G$ be a group. 
\begin{enumerate}
    \item An \hldef{automorphism of $G$} is an \CrefAndHyperrefIfExist{definition:group_homomorphism}{isomorphism} $\phi: G \to G$.

    \item The \hldef{automorphism group of a group $G$,} denoted \hl{$\operatorname{Aut}(G)$}, is the set of all automorphisms of $G$ equipped with the binary operation of function composition. For $\phi, \psi \in \operatorname{Aut}(G)$, the composition $\phi \circ \psi$ is defined by $(\phi \circ \psi)(g) = \phi(\psi(g))$ for all $g \in G$.
\end{enumerate}
\TextIfExists{definition:automorphism_of_an_object_of_a_category_automorphism_group_of_an_object_of_a_locally_small_category}{The automorphism group of a group $G$ is the \CrefAndHyperref{definition:automorphism_of_an_object_of_a_category_automorphism_group_of_an_object_of_a_locally_small_category}{automorphism group} of $G$ as an object of the \CrefAndHyperrefIfExist{definition:category_of_groups_of_abelian_groups}{category of groups}.}
\end{definition}
