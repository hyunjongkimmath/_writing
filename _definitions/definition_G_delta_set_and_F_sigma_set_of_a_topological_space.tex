\begin{definition} \label{definition:G_delta_set_and_F_sigma_set_of_a_topological_space}
Let $(X,\mathcal{T})$ be a \CrefAndHyperrefIfExist{definition:topological_space}{topological space} and let $A \subseteq X$.
\begin{itemize}
  \item The set $A$ is called a \hldef{$G_\delta$ set} if there exists a countable collection of open sets $\{U_n\}_{n \in \mathbb{N}} \subseteq \mathcal{T}$ such that
  $$\hlin{A = \bigcap_{n \in \mathbb{N}} U_n}.$$
  \item The set $A$ is called an \hldef{$F_\sigma$ set} if there exists a countable collection of closed sets $\{F_n\}_{n \in \mathbb{N}}$ such that
  $$\hlin{A = \bigcup_{n \in \mathbb{N}} F_n}.$$
\end{itemize}
\end{definition}
