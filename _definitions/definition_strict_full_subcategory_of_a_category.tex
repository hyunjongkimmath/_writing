
\begin{definition} \label{definition:strict_full_subcategory_of_a_category}
Let $\calD$ be a \CrefAndHyperrefIfExist{definition:category}{(large) category}. A \CrefAndHyperrefIfExist{definition:full_subcategory_of_a_category}{full subcategory} $\mathcal{D} \subseteq \mathcal{C}$ is called a \hldef{strict full subcategory} if it is \CrefAndHyperrefIfExist{definition:replete_subcategory_of_a_category}{replete}. %additionally
% $$\mathrm{Ob}(\mathcal{D}) \subseteq \mathrm{Ob}(\mathcal{C})$$
% is literally (set-theoretically) a subclass of $\mathrm{Ob}(\mathcal{C})$, and not merely identified with it up to bijection. 
\end{definition}
