
\begin{definition}[Topological fields]  \label{definition:topological_vector_space_over_a_topological_field}
    \TODO{Product topology}
    Let $(F,+,\cdot)$ be a \CrefAndHyperrefIfExist{definition:topological_field}{topological field}. A \hldef{topological vector space over $F$} or an \hldef{$F$-topological vector space} is an $F$-vector space $(V,+,\cdot)$ together with a \CrefAndHyperrefIfExist{definition:topological_space}{topology} $\mathcal{T}_V$ on $V$ such that the maps
    \begin{align*}
    +&: V \times V \to V  \quad (v,w) \mapsto v+w \\ 
    \cdot&: F \times V \to V  \quad (a,v) \mapsto a \cdot v \\ 
    -&: V \to V  \quad v \mapsto -v \\
    \end{align*}
    are \CrefAndHyperrefIfExist{definition:continuous_map_between_open_subsets_of_euclidean_spaces}{continuous} with respect to the product topologies on $V \times V$ and $F \times V$ and the topology $\mathcal{T}_V$ on $V$. Equivalently, a topological vector space is a vector space over a topological field with a topology which becomes a \CrefAndHyperrefIfExist{definition:topological_group}{topological group} under the additive structure and whose scalar multiplication map is continuous.
\end{definition}
