\begin{definition} \label{definition:fiber_of_a_morphism_of_schemes_at_a_point}
Let $f: X \to Y$ be a \CrefAndHyperrefIfExist{definition:morphism_of_schemes}{morphism of schemes}. 
For any point $y \in Y$, we define the \hldef{fiber of $f$ at $y$}, denoted \hl{$X_y$}, as the scheme-theoretic \CrefAndHyperrefIfExist{definition:cartesian_product_of_two_objects_in_a_category_over_an_object}{fiber product}:
$$X_y := X \times_Y \operatorname{Spec}(\kappa(y))$$
where $\kappa(y)$ is the \CrefAndHyperrefIfExist{definition:residue_field_of_a_scheme_at_a_point}{residue field} of $Y$ at $y$. Explicitly, the underlying topological space of $X_y$ is the subspace \CrefAndHyperrefIfExist{definition:fiber_of_a_map_of_topological_spaces_over_a_point}{$f^{-1}(y) \subset X$} equipped with the induced topology.
\end{definition}
