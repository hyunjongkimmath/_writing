\begin{definition}[Isometry of quadratic modules] \label{definition:isometry_of_quadratic_modules_over_an_involution_ring}
Let $R$ be a \CrefAndHyperrefIfExist{definition:involution_on_a_ring}{ring with involution} and $(M,q)$, $(M',q')$ \CrefAndHyperrefIfExist{definition:quadratic_form_over_a_ring_with_involution}{quadratic modules} over $R$ (right or left modules consistently).

An \hldef{isometry} $\phi: M \to M'$ is an \CrefAndHyperrefIfExist{definition:homomorphism_of_modules_over_a_ring}{$R$-module isomorphism} such that $q'( \phi(m) ) = q(m)$ for all $m \in M$.

Two quadratic modules $(M,q)$ and $(M',q')$ are \hldef{\emph{isometric}}, written \hl{$(M,q) \cong (M',q')$}, if there exists an isometry $\phi: M \to M'$.

% Equivalently, $\phi$ preserves the \CrefAndHyperrefIfExist{definition:quadratic_form_over_a_ring_with_involution}{polarization}: $B_{q'}(\phi(x),\phi(y)) = B_q(x,y)$ for all $x,y \in M$, since $q$ determines $B_q$ and vice versa over characteristic not $2$.
\end{definition}
