\begin{definition} \label{definition:open_and_closed_subscheme_of_a_scheme}
Let $X$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme}.
\begin{itemize}
    \item An \hldef{open subscheme of $X$} is a subscheme $U$ whose structure morphism $i: U \to X$ is an \CrefAndHyperrefIfExist{definition:open_and_closed_immersions_of_schemes}{open immersion}. The underlying topological space of $U$ is an open subset of $X$, and its structure sheaf is the restriction of $\mathcal{O}_X$ to $U$.
    \item A \hldef{closed subscheme of $X$} is a subscheme $Z$ whose structure morphism $i: Z \to X$ is a \CrefAndHyperrefIfExist{definition:open_and_closed_immersions_of_schemes}{closed immersion}. The underlying topological space of $Z$ is a closed subset of $X$, and the morphism $i^\sharp: \mathcal{O}_X \to i_* \mathcal{O}_Z$ is surjective.
\end{itemize}
\end{definition}
