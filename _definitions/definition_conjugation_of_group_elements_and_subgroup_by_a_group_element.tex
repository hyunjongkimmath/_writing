
\begin{definition}[Conjugation] \label{definition:conjugation_of_group_elements_and_subgroup_by_a_group_element}
Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group}. 
\begin{enumerate}
    \item For $g,h \in G$, the \hldef{conjugate of $h$ by $g$} is the element $g h g^{-1} \in G$.
    Some may choose an opposite convention, letting the conjugate of $h$ by $g$ refer to the element $g^{-1} h g$ instead.

    \item For $g \in G$ and a \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup} $H \leq G$, the \hldef{conjugate of $H$ by $g$} is the group 
    $$\hlin{gHg^{-1} \coloneq \{ghg^{-1} \in G: h \in G}.$$
    It is a subgroup of $G$.

    Some may choose an opposite convention, letting the conjugate of $H$ by $g$ to refer to the group $g^{-1} H g$ instead.
\end{enumerate}
\end{definition}
