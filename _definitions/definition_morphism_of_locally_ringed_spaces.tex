
\begin{definition}[Morphism of locally ringed spaces] \label{definition:morphism_of_locally_ringed_spaces}
    Let $(X, \mathcal{O}_X)$ and $(Y, \mathcal{O}_Y)$ be \CrefAndHyperrefIfExist{definition:locally_ringed_space_on_a_topological_space}{locally ringed spaces}.  
    A \hldef{morphism of locally ringed spaces} is a \CrefAndHyperrefIfExist{definition:morphism_of_ringed_spaces}{morphism of ringed spaces} $(f, f^\#): (X, \mathcal{O}_X) \to (Y, \mathcal{O}_Y)$ such that for each $x \in X$, the \CrefAndHyperrefIfExist{definition:induced_ring_homomorphism_of_stalks_for_a_morphism_of_ringed_spaces}{induced ring homomorphism} of \CrefAndHyperrefIfExist{definition:stalk_of_a_presheaf_on_a_topological_space_at_a_point}{stalks}
    $$f_x^\#: \mathcal{O}_{Y, f(x)} \to \mathcal{O}_{X, x}$$
    is a \CrefAndHyperrefIfExist{definition:local_morphism_of_local_rings}{local homomorphism of local rings}; that is, $(f_x^\#)^{-1}(\mathfrak{m}_x) = \mathfrak{m}_{f(x)}$ where $\mathfrak{m}_x \subseteq \mathcal{O}_{X, x}$ and $\mathfrak{m}_{f(x)} \subseteq \mathcal{O}_{Y, f(x)}$ are the maximal ideals.
\end{definition}
