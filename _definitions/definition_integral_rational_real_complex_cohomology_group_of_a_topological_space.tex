
\begin{definition}
    Let $X$ be a \CrefAndHyperrefIfExist{definition:topological_space}{topological space}. The \CrefAndHyperrefIfExist{definition:singular_cohomology_group_relative_singular_cohomology_group_of_topological_space_with_coefficients_in_a_commutative_ring}{singular cohomology groups} $H^k(X, \bbZ)$, $H^k(X, \bbQ)$, $H^k(X, \bbR)$ $H^k(X, \bbC)$ are called the \hldef{$k$-th integral, rational, real, and complex (singular) cohomology groups of $X$} respectively.
\end{definition}
