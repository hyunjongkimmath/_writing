\begin{definition} \label{definition:residue_field_of_a_scheme_at_a_point}
    Let $X$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme} and let $x \in X$ be a point. The \hldef{residue field of the scheme $X$ at the point $x$} is defined as
    $$\hlin{\kappa(x) = \kappa(\mathcal{O}_{X,x}) := \mathcal{O}_{X,x} / \mathfrak{m}_x.}$$
    where \CrefAndHyperrefIfExist{definition:stalk_of_a_presheaf_on_a_topological_space_at_a_point}{$\mathcal{O}_{X,x}$} is the local ring of $X$ at $x$ with maximal ideal $\mathfrak{m}_x$. 
\end{definition}
