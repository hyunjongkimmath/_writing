\begin{definition} \label{definition:sesquilinear_form_associated_to_a_homomorphism_from_a_module_to_its_twisted_dual}
    Let $(A,\sigma)$ be a \CrefAndHyperref{definition:involution_on_a_ring}{ring with involution}, and let $M$ be a \CrefAndHyperref{definition:module_of_a_ring}{(left) $A$-module}. Let $h: M \to M^*$ be a (left) $A$-module homomorphism where $M^*$ is the \CrefAndHyperref{definition:twisted_dual_module_of_a_module_over_a_ring_with_involution}{twisted dual module of $M$}. The \hldef{sesquilinear form associated to $h$} is \hl{$b_h: M \times M \to A$} given by $b_h(x,y) = h(x)(y)$. 
\end{definition}
