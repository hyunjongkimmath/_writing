\begin{definition} \label{definition:homogeneous_prime_and_maximal_ideal_of_a_graded_ring_by_a_monoid}
Let $R$ be a \CrefAndHyperrefIfExist{definition:graded_ring_by_a_monoid}{(not necessarily commutative) graded ring}.
\begin{itemize}
    \item A \hldef{homogeneous prime ideal of $R$} a \CrefAndHyperrefIfExist{definition:prime_and_maximal_ideal_of_a_ring}{(necessarily two-sided) prime ideal} that is \CrefAndHyperrefIfExist{definition:homogeneous_ideal_of_a_graded_ring_by_a_monoid}{homogeneous}.

    \item A \hldef{homogeneous maximal ideal of $R$} is a \CrefAndHyperrefIfExist{definition:homogeneous_ideal_of_a_graded_ring_by_a_monoid}{homogeneous} proper (left/right/two-sided) ideal  $\mathfrak{m} \subsetneq R$ such that there are no homogeneous (left/right/two-sided) ideals strictly contained between $\mathfrak{m}$ and $R$.
\end{itemize}
\end{definition}
