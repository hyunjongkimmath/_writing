
\begin{definition}[Principal ideal ring/domain (PID)] \label{definition:principal_ideal_ring}
Let $R$ be a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative unital ring}.
Then $R$ is a \hldef{principal ideal ring (PIR)} if every ideal of $R$ is \CrefAndHyperref{definition:ideal_generated_by_a_subset_of_a_ring}{principal}.
If $R$ is additionally an \CrefAndHyperref{definition:zero_divisor_of_a_ring}{integral domain}, then $R$ is said to be a \hldef{principal ideal domain (PID)}
\end{definition}
