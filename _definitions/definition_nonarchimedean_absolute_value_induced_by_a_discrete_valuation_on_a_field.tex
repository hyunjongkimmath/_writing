\begin{definition} \label{definition:nonarchimedean_absolute_value_induced_by_a_discrete_valuation_on_a_field}
    Let $K$ be a field and let $v: K \to \bbZ$ be a \hyperrefIfExists{definition:discrete_valuation_on_a_field}{discrete valuation}\CrefIfExists{definition:discrete_valuation_on_a_field}. If a positive real number $c < 1$. The function
    $$\hlin{|x|_v := \begin{cases} c^{v(x)} & \text{if } x \in K^\times \\ 0 &\text{if } x = 0 \end{cases}}$$
    is a \hyperrefIfExists{definition:archimedean_absolute_value_on_a_field}{nonarchimedean absolute value on $K$}\CrefIfExists{definition:archimedean_absolute_value_on_a_field}. Different values of $c$ yield \hyperrefIfExists{definition:equivalent_absolute_values_on_a_field}{equivalent absolute values} and thus do not change the \hyperrefIfExists{definition:topology_on_a_field_induced_by_an_absolute_value}{topology induced by $|\cdot|_v$}\CrefIfExists{definition:topology_on_a_field_induced_by_an_absolute_value}
\end{definition}
