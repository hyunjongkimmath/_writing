\begin{definition} \label{definition:regular_ring}
Let $A$ be a \CrefAndHyperrefIfExist{definition:noetherian_ring}{Noetherian ring}. We say that $A$ is a \hldef{regular ring} if for every \CrefAndHyperrefIfExist{definition:prime_and_maximal_ideal_of_a_ring}{prime ideal} $\mathfrak{p} \subset A$, the \CrefAndHyperrefIfExist{definition:localization_of_a_commutative_ring_by_a_multiplicative_subset}{localization} $A_{\mathfrak{p}}$ is a \CrefAndHyperrefIfExist{definition:embedding_dimension_of_a_noetherian_local_ring_and_a_regular_local_ring}{regular local ring}.
\end{definition}
