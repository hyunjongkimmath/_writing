\begin{definition}[Isotropic and hyperbolic] \label{definition:isotropic_vector_subspace_anisotropic_quadratic_form_of_a_module_over_a_commutative_ring_with_quadratic_form}
Let $R$ be a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative ring}. Let $q: M \to R$ be a \CrefAndHyperrefIfExist{definition:quadratic_formm_on_a_module_over_a_commutative_ring}{quadratic form} on a \CrefAndHyperrefIfExist{definition:module_of_a_ring}{module} over $R$.
\begin{enumerate}
\item An element $x \in M$ is \hldef{\emph{isotropic}} if $q(x) = 0$.
\item A subspace $L \subset M$ is \hldef{\emph{isotropic}} if $q(L) = 0$.
\item The quadratic form $q$ is \hldef{\emph{anisotropic}} if it has no non-zero isotropic vectors, i.e., $q(x) = 0$ implies $x = 0$.
\end{enumerate}
\end{definition}
