\begin{definition}[Cosets] \label{definition:coset_of_a_subgroup_of_a_group_by_an_element}
Let $G$ be a \CrefAndHyperrefIfExist{definition:group}{group} and $H \le G$ a \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup}.  
For each element $g \in G$, the \hldef{left coset of $H$ in $G$ determined by $g$} is
$$ \hlin{gH := \{gh \mid h \in H\}}, $$
and the \hldef{right coset of $H$ in $G$ determined by $g$} is
$$ \hlin{Hg := \{hg \mid h \in H\}}.  $$
\end{definition}
