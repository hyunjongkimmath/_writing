
\begin{definition}[Normalizer] \label{definition:normalizer_of_a_subgroup_of_a_group}
Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} and let $H \subseteq G$ be a \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup}. The \hldef{normalizer of $H$ in $G$} is defined as
$$\hlin{N_G(H) = \{ g \in G : g H g^{-1} = H \}}.$$
\end{definition}
