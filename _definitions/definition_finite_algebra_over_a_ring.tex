
\begin{definition} \label{definition:finite_algebra_over_a_ring}
    Let $B \to A$ be a \CrefAndHyperrefIfExist{definition:algebra_of_a_ring}{$B$-algebra}. We say that $A$ is a \hldef{finite $B$-algebra} if $A$ is \CrefAndHyperrefIfExist{definition:finitely_generated_algebra_over_a_not_necessarily_commutative_ring}{finitely generated} as a $B$-module.
\end{definition}
