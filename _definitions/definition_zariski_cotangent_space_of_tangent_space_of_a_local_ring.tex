\begin{definition} \label{definition:zariski_cotangent_space_of_tangent_space_of_a_local_ring}
Let $(A, \mathfrak{m})$ be a \CrefAndHyperrefIfExist{definition:locally_ringed_space_on_a_topological_space}{local ring} with \CrefAndHyperrefIfExist{definition:residue_field_of_a_ring_at_a_point}{residue field} $k = A/\mathfrak{m}$.
\begin{itemize}
    \item The \hldef{Zariski cotangent space} of $A$ is defined as the $k$-vector space $\mathfrak{m}/\mathfrak{m}^2$\CrefIfExists{definition:quotient_module_of_a_module_by_a_module_of_a_ring}\CrefIfExists{definition:product_of_ideals_of_a_ring}.

    \item The \hldef{Zariski tangent space} (or simply \hldef{tangent space}) of $A$, denoted by \hl{$T_A$} or \hl{$T_{\mathfrak{m}}(A)$}, is defined as the \CrefAndHyperrefIfExist{definition:dual_of_a_left_right_two_sided_module}{dual vector space} over $k$ to the cotangent space:
    $$\hlin{T_A = \operatorname{Hom}_k(\mathfrak{m}/\mathfrak{m}^2, k).}$$
\end{itemize}
\end{definition}
