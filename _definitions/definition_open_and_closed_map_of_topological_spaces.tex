\begin{definition} \label{definition:open_and_closed_map_of_topological_spaces}
Let $f: X \to Y$ be a \CrefAndHyperrefIfExist{definition:continuous_map_between_open_subsets_of_euclidean_spaces}{map of topological spaces}.
\begin{itemize}
    \item The map $f$ is called an \hldef{open map} if for every open subset $U \subseteq X$, the image $f(U)$ is an open subset of $Y$.
    \item The map $f$ is called a \hldef{closed map} if for every closed subset $Z \subseteq X$, the image $f(Z)$ is a closed subset of $Y$.
\end{itemize}
\end{definition}
