\begin{definition} \label{definition:linear_combination_of_elements_in_a_module}
% Let $R$ be a \CrefAndHyperrefIfExist{definition:ring}{(not necessarily commutative) ring}. 
Let $R,S$ be \CrefAndHyperrefIfExist{definition:ring}{(not-necessarily commutative) rings}. 
\begin{enumerate}
    \item Let $M$ be a \CrefAndHyperrefIfExist{definition:module_of_a_ring}{$R$-$S$-bimodule}. Given elements $x_1,\ldots,x_n \in M$, $r_1,\ldots,r_n \in R$, and $s_1,\ldots,s_n \in R$, an element of the form
    $$\sum_{i=1}^n r_i x_i s_i \in M$$
    is called a \hldef{($R$-$S$-)linear combination of $x_1,\ldots,x_n$}. Given an arbitrary subset $\{m_i\}_{i \in I} \subseteq M$, a \hldef{($R$-$S$-)linear combination of the $m_i$ over $R$} refers to a linear combination of some finite subset of $\{m_i\}_{i \in I}$ over $R$. 

    \item Let $M$ be a \CrefAndHyperrefIfExist{definition:module_of_a_ring}{left/right/two-sided $R$-module}. A \hldef{llinear combination of elements $\{m_i\}_{i \in I} \subseteq M$} refers to a linear combination of (a finite subset of) $\{m_i\}_{i \in I}$ for the \CrefAndHyperrefIfExist{definition:module_of_a_ring}{natural bimodule structure} of $M$. In case that $M$ is a left/right $R$-module, and given finitely many $m_1,\ldots,m_n \in M$, a linear combination of these $m_i$ is equivalently an element of the form
    $$r_1 x_1 + r_2 x_2 + \cdots + r_n x_n$$
    $$x_1 r_1 + x_2 r_2 + \cdots + x_n r_n$$
    for $r_1,\ldots,r_n \in R$ respectively.
\end{enumerate}
