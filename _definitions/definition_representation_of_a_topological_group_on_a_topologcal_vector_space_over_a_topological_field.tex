
\begin{definition} \label{definition:representation_of_a_topological_group_on_a_topologcal_vector_space_over_a_topological_field}
    Let $(G,\cdot,\mathcal{T})$ be a \CrefAndHyperrefIfExist{definition:topological_group}{topological group} and let $V$ be a \CrefAndHyperrefIfExist{definition:topological_vector_space_over_a_topological_field}{topological vector space} over a \CrefAndHyperrefIfExist{definition:topological_field}{topological field} $k$. Let $G$ be a \CrefAndHyperrefIfExist{definition:topological_group}{topological group}. \TODO{discuss the topological structure on $\mathrm{GL}(V)$}

    A \hldef{representation of $G$ on $V$} is a continuous group homomorphism
    $$ \rho : G \to \mathrm{GL}(V)$$

    Alternatively, a \hldef{representation of $G$ on $V$} is a group homomorphism
    $$ \rho : G \to \mathrm{GL}(V)$$
    such that the action map 
    $$G \times V \to V, \quad (g,v) \mapsto \rho(g) v$$ 
    is continuous.

    Under mild conditions, these two notions are equivalent. \TODO{verify that the two notions are equivalent.}

    In particular, the above notions are applicable when $G$ is a \CrefAndHyperrefIfExist{definition:profinite_group}{profinite group}.

\end{definition}
