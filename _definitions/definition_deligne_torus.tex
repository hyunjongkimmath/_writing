\begin{definition}[Deligne torus] \label{definition:deligne_torus}
    Let $\mathbb{G}_{m,\mathbb{C}}$ be the \CrefAndHyperrefIfExist{definition:multiplicative_group_scheme_over_a_scheme}{multiplicative group} over $\mathbb{C}$, and let $\mathrm{Res}_{\mathbb{C}/\mathbb{R}}$ denote \CrefAndHyperrefIfExist{definition:weil_restriction_on_schemes}{Weil restriction} of scalars from $\mathbb{C}$ to $\mathbb{R}$.  The \hldef{Deligne torus} is the real \CrefAndHyperrefIfExist{definition:algebraic_torus_over_a_field}{algebraic torus}
    $$\hlin{\mathbb{S} := \mathrm{Res}_{\mathbb{C}/\mathbb{R}} \mathbb{G}_{m,\mathbb{C}}.}$$
    It satisfies $\mathbb{S}(\mathbb{R}) \cong \mathbb{C}^\times$, and base change to $\mathbb{C}$ yields $\mathbb{S}_{\mathbb{C}} \cong \mathbb{G}_{m,\mathbb{C}} \times \mathbb{G}_{m,\mathbb{C}}$.
\end{definition}
