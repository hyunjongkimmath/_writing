\begin{definition} \label{definition:alternation_of_an_integral_noetherian_scheme}
    \TODO{integral, noeterhain scheme, regular, proper, dominant, function fields, generically finite}
Let $X$ be an integral \CrefAndHyperrefIfExist{}{noetherian scheme}. An \hldef{alteration of $X$} is a \CrefAndHyperrefIfExist{definition:morphism_of_schemes}{morphism of schemes} $f \colon X' \to X$ satisfying the following conditions:
\begin{itemize}
    \item The scheme $X'$ is integral and regular.
    \item The morphism $f$ is proper and dominant.
    \item The extension of function fields $[k(X') : k(X)]$ induced by $f$ is finite; i.e., $f$ is generically finite.
\end{itemize}
\end{definition}
