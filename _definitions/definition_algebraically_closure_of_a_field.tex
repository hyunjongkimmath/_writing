\begin{definition} \label{definition:algebraically_closure_of_a_field}
Let $K$ be a field. A field extension $L / K$ is called an \hldef{algebraic closure of $K$} if the following two conditions hold:
\begin{enumerate}
    \item Every element $a \in L$ is \CrefAndHyperrefIfExist{definition:algebraic_element_minimal_polynomial_of_an_algebraic_element_algebraic_field_extension_transcendental_element_field_extension}{algebraic over $K$}.
    \item The field $L$ is \CrefAndHyperrefIfExist{definition:algebraically_closed_field}{algebraically closed}.
\end{enumerate}
One often writes an algebraic closure of $K$ by \hl{$\overline{K}$}.
\end{definition}
