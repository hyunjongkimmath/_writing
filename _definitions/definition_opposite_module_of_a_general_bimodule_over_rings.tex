\begin{definition} \label{definition:opposite_module_of_a_general_bimodule_over_rings}
    Let $R$ and $S$ be \CrefAndHyperrefIfExist{definition:ring}{rings}. Let $M$ be an \CrefAndHyperrefIfExist{definition:module_of_a_ring}{$R$-$S$-bimodule}. 
    The \hldef{opposite module} \hl{$M^{op}$} is the abelian group $M$ viewed as an $S^{op}$-$R^{op}$-bimodule. For $s \in S^{op}$\CrefIfExists{definition:opposite_ring_of_a_ring}, $r \in R^{op}$, and $m \in M^{op}$, the left $S^{op}$-action and right $R^{op}$-action are defined by:
    $$s \cdot m = ms$$
    $$m \cdot r = rm$$
    where the operations on the right-hand side are the original actions of $S$ and $R$ on $M$.
\end{definition}
