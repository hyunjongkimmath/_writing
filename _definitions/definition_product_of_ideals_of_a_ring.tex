
\begin{definition} \label{definition:product_of_ideals_of_a_ring}
    Let $R$ be a ring, and let $I, J$ be \CrefAndHyperrefIfExist{definition:ideal_of_a_ring}{left ideals of $R$} (resp. right ideals, resp. two-sided ideals).

    The \hldef{product ideal} \hl{$I J$} is the left (resp. right, resp. two-sided) ideal defined as the additive \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup} of $R$ \CrefAndHyperrefIfExist{definition:ideal_generated_by_a_subset_of_a_ring}{generated by} all products $xy$ where $x \in I$ and $y \in J$. In other words,
    \begin{align*}
        IJ = \left \{ \sum_{k=1}^n x_ky_k : n \geq 0, x_k \in I, y_k \in J \right \}.
    \end{align*}

    More generally, we may speak of the product of finitely many ideals of $R$.
\end{definition}
