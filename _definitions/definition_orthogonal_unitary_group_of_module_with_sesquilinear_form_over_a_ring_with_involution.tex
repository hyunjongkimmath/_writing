\begin{definition} \label{definition:orthogonal_unitary_group_of_module_with_sesquilinear_form_over_a_ring_with_involution}
    Let $(A, \sigma)$ be a \CrefAndHyperrefIfExist{definition:involution_on_a_ring}{ring with involution}. Let $(M,b)$ be \CrefAndHyperref{definition:module_of_a_ring}{(right) module} equipped with a \CrefAndHyperref{definition:sesquilinear_form_on_a_module_over_a_ring_with_involution}{sesquilinear form}.
    The \CrefAndHyperref{definition:isometry_between_modules_with_sesquilinear_forms_over_ring_with_involution}{isometries} of $(M,b)$ form a group with respect to composition called the \hldef{orthgonal} or \hldef{unitary group of (M,b)}. The group may be denoted by notations such as \hl{$O(M,b)$}, \hl{$O_A(M,b)$}, \hl{$U(M,b)$}, and \hl{$U_A(M,b)$}.
\end{definition}
