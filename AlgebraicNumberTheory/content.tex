\begin{theorem} \label{theorem:classification_of_local_fields}
    \TODO{TODO: define Laurent seires}
Up to isomorphism, \hyperrefIfExists{definition:local_field}{local fields}\CrefIfExists{definition:local_field} consists precisely of:
\begin{itemize}
    \item $\mathbb{R}$ and $\mathbb{C}$, which are the \hyperrefIfExists{definition:archimedean_absolute_value_on_a_field}{archimedean} local fields),
    \item finite extensions of $\mathbb{Q}_p$ for a prime $p$, which are the nonarchimedean local fields of characteristic $0$, and
    \item finite extensions of $\mathbb{F}_p((t))$, the field of formal Laurent series over a finite field, which are the non-archimedean local fields of positive characteristic.
\end{itemize}
\end{theorem}

\section{Chebotarev density theorem}

The Chebotarev density theorem is a statement that Frobenius elements $\mathrm{Frob}_{\mathfrak{p}}$ are equidistributed across the conjugacy classes of $\operatorname{Gal}(L/K)$ of a finite Galois extension of number fields, showing that the splitting behavior of primes in extensions is governed by uniform distribution with respect to the Galois group structure.

\begin{theorem}[Chebotarev Density Theorem for number fields]
    \TODO{define unramified prime}
Let $L/K$ be a finite \CrefAndHyperrefIfExist{definition:galois_extension_of_fields}{Galois extension} of \CrefAndHyperrefIfExist{definition:global_field}{number fields} with \CrefAndHyperrefIfExist{definition:galois_extension_of_fields}{Galois group $G = \mathrm{Gal}(L/K)$}. For a conjugacy class $C \subseteq G$, let 
\[ \pi_C(x) = \#\{ \, \mathfrak{p} \subseteq \mathcal{O}_K : N\mathfrak{p} \leq x,\ \mathfrak{p} \text{ unramified in } L,\ \mathrm{Frob}_{\mathfrak{p}} \in C \,\} \]
be the number of prime ideals $\mathfrak{p}$ of $K$ with norm at most $x$ whose Frobenius conjugacy class $\mathrm{Frob}_{\mathfrak{p}}$ in $G$ equals $C$.

\TODO{define natural density of primes}
Then the natural density of such primes exists and satisfies
\[
\lim_{x \to \infty} \frac{\pi_C(x)}{\pi(x)} = \frac{|C|}{|G|},
\]
where $\pi(x)$ is the number of prime ideals $\mathfrak{p} \subseteq \mathcal{O}_K$ with $N\mathfrak{p} \leq x$.
\end{theorem}
