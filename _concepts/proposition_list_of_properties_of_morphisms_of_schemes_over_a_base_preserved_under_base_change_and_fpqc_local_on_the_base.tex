\begin{proposition}[See {\cite[02WH]{stacks-project}}] \label{proposition:list_of_properties_of_morphisms_of_schemes_over_a_base_preserved_under_base_change_and_fpqc_local_on_the_base}
    Let $S$ be a scheme. The following are properties for morphisms of $S$-schemes that are preserved under base change and \CrefAndHyperrefIfExist{definition:tau_local_property_of_a_morphism_of_schemes_over_a_scheme}{fpqc-local on the base}:
    \begin{itemize}
        \item closed immersions
        \item open immersions
        \item quasi-compact immersions
        \item quasi-compact
        \item universally closed
        \item quasi-separated
        \item separated
        \item monomorphism
        \item surjective
        \item universally injective
        \item affine
        \item quasi-affine
        \item locally of finite type
        \item finite type
        \item locally quasi-finite
        \item quasi-finite
        \item locally of finite presentation
        \item finite presentation
        \item locally of finite type of relative dimension $d$
        \item universally open
        \item flat
        \item syntomic
        \item smooth
        \item unramified
        \item $G$-unramified
        \item \'etale
        \item proper
        \item finite
        \item integral
        \item finite locally free
        \item universally submersive
        \item universal homeomorphism
    \end{itemize}
\end{proposition}
