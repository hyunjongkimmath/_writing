\begin{theorem} \label{theorem:morphism_of_schemes_is_finite_if_and_only_if_affine_and_proper_if_and_only_if_quasifinite_and_proper}
    Let $f: X \to Y$ be a morphism of schemes. The following are equivalent:
    \begin{enumerate}
        \item $f$ is \CrefAndHyperrefIfExist{definition:finite_morphism_of_schemes}{finite}.
        \item $f$ is \CrefAndHyperrefIfExist{definition:affine_morphism_of_schemes}{affine} and \CrefAndHyperrefIfExist{definition:proper_morphism_of_schemes}{proper}.
        \item $f$ is \CrefAndHyperrefIfExist{definition:quasifinite_morphism_of_schemes}{quasifinite} and \CrefAndHyperrefIfExist{definition:proper_morphism_of_schemes}{proper}.
    \end{enumerate}
\end{theorem}
