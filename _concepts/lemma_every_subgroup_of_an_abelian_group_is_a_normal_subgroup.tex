\begin{lemma} \label{lemma:every_subgroup_of_an_abelian_group_is_a_normal_subgroup}
    Every \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup} of an \CrefAndHyperrefIfExist{definition:group}{abelian group} is a \CrefAndHyperrefIfExist{definition:normal_subgroup_of_a_group}{normal subgroup}.
\end{lemma}
