
\begin{theorem} \label{theorem:complex_manifold_is_equivalent_to_a_smooth_manifold_of_even_real_dimension_with_an_integrable_almost_complex_structure}
Let $X$ be a \hyperrefIfExists{definition:C_k_manifold}{smooth manifold}of real dimension $2n$. Then the following are equivalent:
\begin{enumerate}
    \item[(1)] $X$ admits the structure of a complex manifold of complex dimension $n$.
    \item[(2)] There exists an \hyperrefIfExists{definition:integrable_almost_complex_structure_on_a_smooth_manifold_of_even_real_dimension}{integrable} \hyperrefIfExists{definition:almost_complex_structure_on_a_smooth_manifold_of_even_real_dimension}{almost complex structure} $J : TX \to TX$ on $X$ 
\end{enumerate}
In other words, a complex manifold structure on $X$ is completely characterized by an integrable almost complex structure on the underlying smooth manifold.
\end{theorem}
