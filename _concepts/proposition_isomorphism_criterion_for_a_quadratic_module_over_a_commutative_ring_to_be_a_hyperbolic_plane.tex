\begin{proposition}[Isomorphism criterion] \label{proposition:isomorphism_criterion_for_a_quadratic_module_over_a_commutative_ring_to_be_a_hyperbolic_plane}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative ring}, $M$ a \CrefAndHyperrefIfExist{definition:module_of_a_ring}{module} of $R$, $H \cong R^2$ a \CrefAndHyperrefIfExist{definition:submodule_of_a_module_over_a_ring}{submodule} of $M$, and $q: M \to R$ a \CrefAndHyperrefIfExist{definition:quadratic_formm_on_a_module_over_a_commutative_ring}{quadratic form}.  Then $(H, q|_H)$ is a \CrefAndHyperrefIfExist{definition:hyperbolic_plane_over_a_commutative_ring}{hyperbolic plane} if and only if $q|_H \cong q_H$ as quadratic forms via some basis of $H$.
\end{proposition}
