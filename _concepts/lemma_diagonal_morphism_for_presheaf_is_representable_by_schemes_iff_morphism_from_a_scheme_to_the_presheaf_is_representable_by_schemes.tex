\begin{lemma}[See {\cite[Tag 025W]{stacks-project}}] \label{lemma:diagonal_morphism_for_presheaf_is_representable_by_schemes_iff_morphism_from_a_scheme_to_the_presheaf_is_representable_by_schemes}
    Let $S$ be a \CrefAndHyperrefIfExist{definition:scheme}{scheme} and let $F$ be a \CrefAndHyperrefIfExist{definition:presheaf_on_a_category}{presheaf} of sets on $\mathit{Sch}/S$. The following are equivalent:
    \begin{enumerate}
        \item The \CrefAndHyperrefIfExist{definition:diagonal_morphism_of_a_morphism_of_schemes}{diagonal morphism} $F \to F \times_S F$ is \CrefAndHyperrefIfExist{definition:representable_by_schemes_for_a_morphism_of_presheaves_on_Sch_S}{representable by schemes}. 
        \item For any object $U$ of $\mathit{Sch}/S$, any morphism \CrefAndHyperrefIfExist{definition:representable_functor_on_a_category_enriched_in_a_monoidal_category}{$h_U \to F$} of presheaves is representable by schemes.
    \end{enumerate}
\end{lemma}
