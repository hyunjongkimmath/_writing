\begin{theorem}[Primitive element theorem] \label{theorem:primitive_element_theorem_of_finite_separable_algebraic_field_extensions}
Let $L/K$ be a \CrefAndHyperrefIfExist{definition:degree_of_a_field_extension}{finite} \CrefAndHyperrefIfExist{definition:separable_algebraic_field_extension}{separable extension}. Then there exists an element $\alpha \in L$ such that 
$$L = K(\alpha).$$
\CrefIfExists{definition:field_generated_by_elements}
In other words, every finite separable extension is \CrefAndHyperrefIfExist{definition:field_generated_by_elements}{simple}.
\end{theorem}
