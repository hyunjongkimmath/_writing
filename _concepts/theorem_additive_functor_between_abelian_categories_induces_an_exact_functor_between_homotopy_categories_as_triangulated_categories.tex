\begin{theorem}[Exact functor induces a functor on homotopy categories, cf. {\cite[Expos\'e XVII, Proposition 1.1.7.]{SGA4_III}}]  \label{theorem:additive_functor_between_abelian_categories_induces_an_exact_functor_between_homotopy_categories_as_triangulated_categories}

Let $F: \prod_{i=1}^n \mathscr{A}_i \to \mathscr{B}$ be an \CrefAndHyperrefIfExist{definition:n_ary_additive_functor_between_additive_categories}{additive multifunctor} of \CrefAndHyperrefIfExist{definition:abelian_category}{abelian categories}. $F$ induces \CrefAndHyperrefIfExist{definition:exact_mixed_variance_multifunctor_from_triangulated_categories_to_a_triangulated_category}{exact multifunctors}\footnote{wlog covariant in each variable, but this can be made multi-variant by taking \CrefAndHyperrefIfExist{definition:opposite_category_of_a_category}{opposite categories}

If $F$ is contravariant in the $j$-th variable, the definitions are adjusted by replacing $\mathscr{A}_j$ with its opposite category $\mathscr{A}_j^{\text{op}}$, thereby interchanging $K^+$ with $K^-$ and $RF$ with $LF$ for that specific variable.}
between the corresponding \CrefAndHyperrefIfExist{definition:homotopy_category_of_chain_complexes_of_an_additive_category}{homotopy categories}:
\begin{align*}
K(\mathscr{A}) &\to K(\mathscr{B}) \\
K^?(\prod \mathscr{A}_i) &\to K^{?}(\mathscr{B}) \\
K^{?,??}(\prod \mathscr{A}_i) &\to K^{?,??}(\mathscr{B})
\end{align*}
(\Cref{definition:homotopy_category_of_chain_complexes_of_an_additive_category}). We write these functors by \hl{$K(F)$}.
% \begin{itemize}
%     \item $K(\prod \mathscr{A}_i) \to K(\mathscr{B})$
%     \item $K^+(\prod \mathscr{A}_i) \to K^+(\mathscr{B})$
%     \item $K^-(\prod \mathscr{A}_i) \to K^-(\mathscr{B})$
% \end{itemize}

% Let $\mathcal{A}$ and $\mathcal{B}$ be \CrefAndHyperrefIfExist{definition:abelian_category}{abelian categories}, and let
% \[
% F: \mathcal{A} \to \mathcal{B}
% \]
% be an \CrefAndHyperrefIfExist{definition:additive_category}{additive functor}. There is a natural induced \CrefAndHyperrefIfExist{definition:exact_functor_between_triangulated_categories}{exact functor} 
% $$\hlin{K(F): K(\calA) \to K(\calB)}$$
% of \CrefAndHyperrefIfExist{definition:triangulated_category}{triangulated categories}(\Cref{proposition:homotopy_category_of_an_additive_category_is_a_triangulated_category}). We also write \hl{$K(F)$} for the exact functors from the triangulated subcategories \CrefAndHyperrefIfExist{definition:homotopy_category_of_chain_complexes_of_an_additive_category}{$K^?(\calA)$ or $K^{?,??}(\calA)$} of $K(\calA)$ to $K(\calB)$. 
\end{theorem}
