\begin{proposition}[{\cite[Proposition 3.1.11(ii), Corollary 3.1.12, Corollary 3.1.13.]{kashiwara_schapira_cs}}] \label{proposition:categories_of_sets_functors_from_small_categories_to_sets_and_modules_satisfy_ipc_property}
    \begin{enumerate}
        \item The \CrefAndHyperrefIfExist{definition:category_of_sets}{category of sets} satisfies the \CrefAndHyperrefIfExist{definition:ipc_property_of_a_category_that_admits_small_products_and_small_filtered_limits}{IPC-property}

        \item Given a \CrefAndHyperrefIfExist{definition:locally_small_category}{small category} $\calC$, the \CrefAndHyperrefIfExist{definition:diagram_in_a_category_indexed_by_a_small_category}{functor category} $[\calC, \Sets]$ satisfies the IPC-property.

        \item Let $R$ be a \CrefAndHyperrefIfExist{definition:category_of_rings}{(not necessarily commutative)} ring. The \CrefAndHyperrefIfExist{definition:category_of_modules_and_bimodules_over_rings}{category of (left) $R$-modules} satisfies the IPC-property; in particular, the \CrefAndHyperrefIfExist{definition:category_of_groups_of_abelian_groups}{category of abelian groups} and the category of \CrefAndHyperrefIfExist{definition:module_of_a_ring}{$R$-$S$-bimodules}, given rings $R$ and $S$, satisfy the IPC-property.
    \end{enumerate}
\end{proposition}
