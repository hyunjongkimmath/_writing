\begin{proposition} \label{proposition:collection_of_hereditarily_finite_sets_is_a_set}
    The collection $V_\omega$ of all \CrefAndHyperrefIfExist{definition:hereditarily_finite_sets}{hereditarily finite sets} is a set and it can be constructed via the cumulative hierarchy by defining $V_0 = \emptyset$, $V_{n+1} = \mathcal{P}(V_n)$ for $n \in \mathbb{N}$, and taking the union $V_\omega = \bigcup_{n < \omega} V_n$.
\end{proposition}
