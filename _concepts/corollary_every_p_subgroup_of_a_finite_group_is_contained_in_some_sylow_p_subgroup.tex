\begin{corollary} \label{corollary:every_p_subgroup_of_a_finite_group_is_contained_in_some_sylow_p_subgroup}
Let $G$ be a finite group and $p$ a prime number.
Every \CrefAndHyperrefIfExist{definition:p_subgroup_of_a_finite_group}{$p$-subgroup} of $G$ is contained in some \CrefAndHyperrefIfExist{definition:sylow_p_subgroup_of_a_finite_group}{Sylow $p$-subgroup} of $G$. 
\end{corollary}
