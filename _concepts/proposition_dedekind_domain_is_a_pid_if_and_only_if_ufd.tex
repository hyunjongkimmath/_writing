
\begin{proposition} \label{proposition:dedekind_domain_is_a_pid_if_and_only_if_ufd}
    Let $R$ be a \CrefAndHyperref{definition:dedekind_domain}{Dedekind domain}. It is a \CrefAndHyperrefIfExist{definition:principal_ideal_ring}{PID} if and only if it is a \CrefAndHyperrefIfExist{definition:unique_factorization_domain}{UFD}.
\end{proposition}
