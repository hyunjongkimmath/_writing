\begin{proposition} \label{proposition:functor_of_points_is_a_sheaf_on_the_big_fpqc_site}
    Let $S$ be a scheme. For every scheme $X/S$, the \CrefAndHyperrefIfExist{proposition:functor_of_points_is_a_sheaf_on_the_big_fpqc_site}{functor of points $h_X$} is a \CrefAndHyperrefIfExist{definition:sheaf_on_a_site}{sheaf} on the big fpqc site $(\mathit{Sch}/S)_{fpqc}$
\end{proposition}
\begin{corollary}
    Let $S$ be a scheme. For every scheme $X/S$, the \CrefAndHyperrefIfExist{proposition:functor_of_points_is_a_sheaf_on_the_big_fpqc_site}{functor of points $h_X$} is a \CrefAndHyperrefIfExist{shea}{}sheaf on $\mathit{Sch}/S$ equipped with any topology weaker than the fpqc topology.
\end{corollary}
\begin{proof}
    This follows from Proposition \ref{proposition:functor_of_points_is_a_sheaf_on_the_big_fpqc_site} as the fppf, \'etale, and Zariski topologies are all coarser than the fpqc topology.
\end{proof}
