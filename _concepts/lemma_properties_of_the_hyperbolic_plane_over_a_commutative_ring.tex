\begin{lemma}[Properties of the hyperbolic plane] \label{lemma:properties_of_the_hyperbolic_plane_over_a_commutative_ring}
Let $(H, q_H)$ be a \CrefAndHyperrefIfExist{definition:hyperbolic_plane_over_a_commutative_ring}{hyperbolic plane} over a \CrefAndHyperrefIfExist{definition:commutative_ring}{commutative ring} $R$. Then:
    \begin{enumerate}
    \item The \CrefAndHyperrefIfExist{definition:quadratic_formm_on_a_module_over_a_commutative_ring}{polarization} is $B_{q_H}((x_1,y_1),(x_2,y_2)) = x_1 y_2 + y_1 x_2$.
    \item The vectors $e_1 = (1,0)$ and $e_2 = (0,1)$ satisfy $q_H(e_1) = q_H(e_2) = 0$ and $B_{q_H}(e_1,e_2) = 1$.

    \item $H$ admits a symplectic basis $\{e_1,e_2\}$ where $B_{q_H}(e_i,e_j) = \delta_{i,3-j}$.
    \end{enumerate}
\end{lemma}
