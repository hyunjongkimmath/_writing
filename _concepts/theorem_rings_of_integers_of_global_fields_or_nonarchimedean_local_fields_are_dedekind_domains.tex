
\begin{theorem} \label{theorem:rings_of_integers_of_global_fields_or_nonarchimedean_local_fields_are_dedekind_domains}
    \begin{enumerate}
        \item The \CrefAndHyperrefIfExist{definition:ring_of_integers_of_a_global_or_local_ring}{ring of integers} of a \CrefAndHyperrefIfExist{definition:nonarchimedean_absolute_value_induced_by_a_discrete_valuation_on_a_field}{nonarchimedean} \CrefAndHyperrefIfExist{definition:local_field}{local field} is a \CrefAndHyperrefIfExist{definition:discrete_valuation_ring}{DVR}. \CrefAndHyperref{proposition:dvrs_are_dedekind_domains}{In particular}, it is a \CrefAndHyperrefIfExist{definition:dedekind_domain}{Dedekind domains}.

        \item The \CrefAndHyperrefIfExist{definition:ring_of_integers_of_a_global_or_local_ring}{ring of integers} of a \CrefAndHyperrefIfExist{definition:global_field}{global field}is a \CrefAndHyperrefIfExist{definition:dedekind_domain}{Dedekind domains}.

    \end{enumerate}
\end{theorem}
