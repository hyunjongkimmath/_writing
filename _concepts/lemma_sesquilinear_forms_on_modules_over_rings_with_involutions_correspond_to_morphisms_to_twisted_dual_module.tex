\begin{lemma} \label{lemma:sesquilinear_forms_on_modules_over_rings_with_involutions_correspond_to_morphisms_to_twisted_dual_module}
    Let $(A,\sigma)$ be a \CrefAndHyperref{definition:involution_on_a_ring}{ring with involution}, and let $M$ be a \CrefAndHyperref{definition:module_of_a_ring}{(left) $A$-module}.
    A \CrefAndHyperref{definition:sesquilinear_form_on_a_module_over_a_ring_with_involution}{sesquilinear form} $b:M \times M \to A$ on $M$ is determined by its \CrefAndHyperref{definition:adjoint_map_of_a_sesquilinear_form_on_a_module_over_a_ring_with_involution}{adjoint} $\hat{b}$. Conversely, $A$-linear homomorphisms $b: M \to M^*$ (\Cref{definition:twisted_dual_module_of_a_module_over_a_ring_with_involution}) are determined by their \CrefAndHyperref{definition:sesquilinear_form_associated_to_a_homomorphism_from_a_module_to_its_twisted_dual}{associated sesquilinear forms} $b_h$. In fact, the map $b \mapsto \hat{b}$ is a (two-sided) \CrefAndHyperref{definition:center_of_a_ring}{$Z(A)$}-module isomorphism
    $$\Sesq_A(M) \cong \Hom_A(M, M^*).$$
\end{lemma}
