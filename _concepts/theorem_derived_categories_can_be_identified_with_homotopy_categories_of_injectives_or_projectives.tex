\begin{theorem}[see e.g. {\cite[Theorem 10.4.8]{weibel}}] \label{theorem:derived_categories_can_be_identified_with_homotopy_categories_of_injectives_or_projectives}
    Let $\calA$ be an abelian category.
    \begin{enumerate}
        \item If $\calA$ \CrefAndHyperrefIfExist{definition:has_enough_injectives_or_projectives_for_an_abelian_category}{has enough injectives}, then the category $D^+(\calA)$\CrefIfExists{definition:derived_category_of_an_abelian_category} is equivalent to the category $K^+(\calI)$ \CrefIfExists{definition:category_of_bounded_complexes_of_injectives_projectives}, which is \hyperrefIfExists{definition:locally_small_category}{locally small}\CrefIfExists{definition:locally_small_category}. In fact, an equivalence is given by 
        $$K^+(\calI) \xrightarrow{q} D^+(\calI) \hookrightarrow D^+(\calA).$$

        \item If $\calA$ \CrefAndHyperrefIfExist{definition:has_enough_injectives_or_projectives_for_an_abelian_category}{has enough projectives}, then the category $D^-(\calA)$\CrefIfExists{definition:derived_category_of_an_abelian_category} is equivalent to the category $K^-(\calP)$ \CrefIfExists{definition:category_of_bounded_complexes_of_injectives_projectives}, which is \hyperrefIfExists{definition:locally_small_category}{locally small}\CrefIfExists{definition:locally_small_category}.
        $$K^-(\calP) \xrightarrow{q} D^-(\calP) \hookrightarrow D^-(\calA).$$
    \end{enumerate}
    
\end{theorem}
