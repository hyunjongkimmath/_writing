
\begin{proposition} \label{proposition:subgroup_of_a_group_is_normal_subgroup_of_normalizer}
    Let $(G,\cdot)$ be a \CrefAndHyperrefIfExist{definition:group}{group} and let $H \subseteq G$ be a \CrefAndHyperrefIfExist{definition:subgroup_of_a_group}{subgroup}. The group $H$ is a \CrefAndHyperrefIfExist{definition:normal_subgroup_of_a_group}{normal subgroup} of the \CrefAndHyperrefIfExist{definition:normalizer_of_a_subgroup_of_a_group}{normalizer $N_G(H)$}.
\end{proposition}
