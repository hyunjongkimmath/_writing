\begin{theorem} \label{theorem:fiber_products_exist_in_the_category_of_schemes_over_a_scheme}
    Let $S$ be a scheme. The category \CrefAndHyperrefIfExist{definition:scheme_over_a_scheme}{$\Sch/S$ of schemes over $S$} is closed under fiber products. More precisely, for any morphisms $X \to Z$ and $Y \to Z$ in $\Sch/S$, there exists a \CrefAndHyperrefIfExist{definition:cartesian_product_of_two_objects_in_a_category_over_an_object}{fiber product $X \times_Z Y$}.
\end{theorem}
