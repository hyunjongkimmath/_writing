\begin{theorem} \label{theorem:some_examples_of_qcqs_schemes}
The following classes of schemes are \CrefAndHyperrefIfExist{definition:quasi_compact_quasi_separated_coherent_scheme}{coherent} (qcqs):
\begin{enumerate}
    \item Any \CrefAndHyperrefIfExist{definition:affine_scheme}{affine scheme} $X = \operatorname{Spec}(R)$ for any commutative ring $R$.
    \item Any \CrefAndHyperrefIfExist{definition:locally_noetherian_and_noetherian_scheme}{Noetherian scheme}.
    \item Any scheme of \CrefAndHyperrefIfExist{definition:finite_type_morphism_of_schemes}{finite type} over a coherent base (e.g., \CrefAndHyperrefIfExist{definition:algebraic_variety_over_a_field}{varieties} over a field).
\end{enumerate}
\end{theorem}
