\begin{corollary}[Dimension formula] \label{corollary:dimension_formula_for_nondegenerate_quadratic_forms_on_vector_spcaes}
Let $R$ be a \CrefAndHyperrefIfExist{definition:field}{field} and $q: M \to R$ a \CrefAndHyperrefIfExist{definition:nondegenerate_quadratic_form_on_a_module_over_a_commutative_ring}{non-degenerate} \CrefAndHyperrefIfExist{definition:quadratic_formm_on_a_module_over_a_commutative_ring}{quadratic form}.
We have $\dim M \equiv \dim M_a \pmod{2}$.
\end{corollary}
