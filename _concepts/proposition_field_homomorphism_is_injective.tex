\begin{proposition} \label{proposition:field_homomorphism_is_injective}
Let $K$ and $L$ be \CrefAndHyperrefIfExist{definition:field}{fields} and let $\phi: K \to L$ be a \CrefAndHyperrefIfExist{definition:field_homomorphism}{field homomorphism}.
Then $\phi$ is \CrefAndHyperrefIfExist{definition:injective_surjective_bijective_map_of_sets}{injective}.
\end{proposition}
