
\begin{theorem}[Correspondence between DVRs and discretely valued fields] \label{theorem:correspondence_between_dvrs_and_discretely_valued_fields}
    \TODO{define field of fractions}
    Let $K$ be a field equipped with a \CrefAndHyperrefIfExist{definition:discrete_valuation_on_a_field}{discrete valuation} $v: K^\times \to \mathbb{Z}$. The \CrefAndHyperrefIfExist{definition:valuation_ring_of_a_valued_field}{valuation ring}
    $$R_v = \{ x \in K \mid v(x) \geq 0 \}$$
    is a \CrefAndHyperrefIfExist{definition:discrete_valuation_ring}{discrete valuation ring}. Conversely, if $(R, \mathfrak{m})$ is a discrete valuation ring with field of fractions $K = \mathrm{Frac}(R)$, then there exists a discrete valuation
    $$v: K^\times \to \mathbb{Z}$$
    such that $R = R_v$ is the valuation ring associated to $v$.

    In particular, the category of discrete valuation rings (up to isomorphism) corresponds bijectively to the category of fields equipped with a discrete valuation.
\end{theorem}
