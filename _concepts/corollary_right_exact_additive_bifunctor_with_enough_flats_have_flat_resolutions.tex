\begin{corollary} \label{corollary:right_exact_additive_bifunctor_with_enough_flats_have_flat_resolutions}
Let $\mathcal{A}, \calB, \calC$ be \CrefAndHyperrefIfExist{definition:abelian_category}{abelian categories}, and let $\otimes : \mathcal{A} \times \mathcal{B} \to \mathcal{C}$ be a bifunctor that is \CrefAndHyperrefIfExist{definition:additive_functor_between_additive_categories}{additive} and \CrefAndHyperrefIfExist{definition:exact_functor_between_abelian_categories}{right exact} in each variable.
\begin{enumerate}
    \item If \CrefAndHyperrefIfExist{definition:has_enough_flat_objects_for_an_abelian_category_with_respect_to_a_right_exact_bifunctor}{$\calA$ has enough flat objects with respect to $\otimes$}, then every object of $A$ has a \CrefAndHyperrefIfExist{definition:flat_resolution_of_an_object_in_an_abelian_category_with_respect_to_a_right_exact_monoidal_product_functor}{flat resolution}.
    \item If \CrefAndHyperrefIfExist{definition:has_enough_flat_objects_for_an_abelian_category_with_respect_to_a_right_exact_bifunctor}{$\calB$ has enough flat objects with respect to $\otimes$}, then every object of $B$ has a \CrefAndHyperrefIfExist{definition:flat_resolution_of_an_object_in_an_abelian_category_with_respect_to_a_right_exact_monoidal_product_functor}{flat resolution}.
\end{enumerate}
\end{corollary}
\begin{proof}
    This follows immediately from \Cref{lemma:an_object_of_abelian_category_with_enough_objects_of_a_class_on_the_right_left_has_right_left_resolution_by_the_class}.
\end{proof}
