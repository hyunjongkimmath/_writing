\begin{lemma}  \label{lemma:category_of_rings_is_category_of_Z_rings}
    The \CrefAndHyperrefIfExist{definition:category_of_rings}{category of rings} is \CrefAndHyperrefIfExist{definition:equivalence_of_categories}{equivalent} to the \CrefAndHyperrefIfExist{definition:category_of_rings_over_a_ring}{category of $\bbZ$-rings}.
\end{lemma}
