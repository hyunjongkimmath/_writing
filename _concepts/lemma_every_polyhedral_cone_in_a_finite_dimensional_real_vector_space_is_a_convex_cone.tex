\begin{lemma} \label{lemma:every_polyhedral_cone_in_a_finite_dimensional_real_vector_space_is_a_convex_cone}
Let $V$ be a \CrefAndHyperrefIfExist{definition:basis_of_a_vector_space_over_a_field}{finite-dimensional} real \CrefAndHyperrefIfExist{definition:vector_space_over_a_field}{vector space}. 
Every \CrefAndHyperrefIfExist{definition:polyhedral_cone_in_a_finite_dimensional_real_vector_space}{polyhedral cone} $\sigma \subset V$ is a \CrefAndHyperrefIfExist{definition:convex_cone_in_a_finite_dimensional_real_vector_space}{convex cone}.
\end{lemma}
