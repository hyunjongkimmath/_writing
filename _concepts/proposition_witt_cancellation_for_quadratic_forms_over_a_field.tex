\begin{proposition}[Witt cancellation] \label{proposition:witt_cancellation_for_quadratic_forms_over_a_field}
    Let $R$ be a \CrefAndHyperrefIfExist{definition:field}{field} and $q: M \to R$, $q': M' \to R$ \CrefAndHyperrefIfExist{definition:quadratic_formm_on_a_module_over_a_commutative_ring}{quadratic forms} such that $(M, q) \cong (M', q') \oplus (H, q_H)$ as quadratic modules, where $(H, q_H)$ is a \CrefAndHyperrefIfExist{definition:hyperbolic_plane_over_a_commutative_ring}{hyperbolic plane}. Then \CrefAndHyperrefIfExist{definition:isometry_of_quadratic_modules_over_a_commutative_ring}{$(M, q) \cong (M', q')$}.
\end{proposition}
