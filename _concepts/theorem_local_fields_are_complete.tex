
\begin{theorem} \label{theorem:local_fields_are_complete}
    Let $K$ be a \hyperrefIfExists{definition:local_field}{local field}\CrefIfExists{definition:local_field}. $K$ is complete under the metric \hyperrefIfExists{definition:metric_induced_by_an_absolute_value_on_a_field}{induced by}\CrefIfExists{definition:metric_induced_by_an_absolute_value_on_a_field} its absolute value.
\end{theorem}
