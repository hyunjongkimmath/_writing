\begin{lemma} \label{lemma:category_of_rings_over_a_ring_has_an_initial_object}
    Let $R$ be a not necessarily commutative \CrefAndHyperrefIfExist{definition:ring}{ring}. 
    The category \CrefAndHyperrefIfExist{definition:category_of_rings_over_a_ring}{$\mathbf{Ring}_R$} of rings over $R$ has an \CrefAndHyperrefIfExist{definition:initial_final_zero_objects_of_a_category}{initial object}, namely $R$ equipped with the identity morphism.
\end{lemma}
