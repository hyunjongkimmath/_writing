\begin{proposition} \label{proposition:codimension_of_an_irreducible_closed_subscheme_of_a_scheme_equals_the_dimension_of_the_localization_at_the_generic_point}
    If $X$ is an \CrefAndHyperrefIfExist{definition:irreducible_topological_space}{irreducible} \CrefAndHyperrefIfExist{definition:scheme}{scheme}, and $Y \subseteq X$ is an irreducible closed subscheme, then \CrefAndHyperrefIfExist{definition:codimension_of_an_irreducible_closed_subset_in_a_topological_space}{$\codim(Y,X)$} coincides with $\dim(\mathcal{O}_{X, \eta})$\CrefIfExists{definition:krull_dimension_of_a_commutative_ring}\CrefIfExists{definition:stalk_of_a_presheaf_on_a_topological_space_at_a_point}, where $\eta$ is the \CrefAndHyperrefIfExist{definition:generic_point_of_a_topological_space}{generic point of $Y$}.
\end{proposition}
