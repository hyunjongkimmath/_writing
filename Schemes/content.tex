\section{Hirzebruch Surfaces}

\input{../_definitions/definition_hirzebruch_surface_over_a_field.tex}

\begin{theorem}[Intersection Pairing on $\mathbb{F}_n$]
Let $\mathbb{F}_n$ be the Hirzebruch surface of degree $n \ge 0$. The intersection form on the Picard group $\operatorname{Pic}(\mathbb{F}_n) \cong \mathbb{Z} f \oplus \mathbb{Z} s$ (where $f$ is the fiber class and $s$ is the class of the unique section with $s^2 = -n$) is given by the matrix:
$$
\begin{pmatrix}
f^2 & f \cdot s \\
s \cdot f & s^2
\end{pmatrix}
=
\begin{pmatrix}
0 & 1 \\
1 & -n
\end{pmatrix}.
$$
Consequently, the intersection pairing is unimodular, and the determinant of the intersection form is $-1$.
\end{theorem}

\begin{theorem}[Canonical Divisor of $\mathbb{F}_n$]
The canonical divisor class $K_{\mathbb{F}_n}$ of the Hirzebruch surface $\mathbb{F}_n$ is given in terms of the basis $\{f, s\}$ by the formula:
$$\hlin{K_{\mathbb{F}_n} \sim -2s - (n+2)f.}$$
\end{theorem}

\begin{theorem}[Cohomology of Line Bundles on $\mathbb{F}_n$]
Let $D \equiv a s + b f$ be a divisor on $\mathbb{F}_n$ with $a, b \in \mathbb{Z}$. The dimension of the space of global sections $H^0(\mathbb{F}_n, \mathcal{O}(D))$ is given by:
$$\hlin{h^0(\mathbb{F}_n, \mathcal{O}(as+bf)) = \sum_{i=0}^a \max(0, b - ni + 1), \quad \text{if } a \ge 0.}$$
If $a < 0$, then $h^0(\mathbb{F}_n, \mathcal{O}(D)) = 0$.
\end{theorem}

\begin{proposition}[Curve Classes]
Let $C \subset \mathbb{F}_n$ be an irreducible curve effectively equivalent to $as + bf$.
\begin{itemize}
    \item If $C = s$ (the unique negative section), then $a=1, b=0$.
    \item If $C \neq s$, then $a \ge 0$ and $b \ge na$.
\end{itemize}
This implies that the cone of effective curves (the Mori cone) $\overline{NE}(\mathbb{F}_n)$ is generated by the classes of the negative section $s$ and the fiber $f$.
\end{proposition}



\begin{theorem}[Structure of Hirzebruch Surfaces]
Let $\mathbb{F}_n = \mathbb{P}(\mathcal{O}_{\mathbb{P}^1} \oplus \mathcal{O}_{\mathbb{P}^1}(n))$ be a Hirzebruch surface over an algebraically closed field $k$, where $n \ge 0$. Let $\pi: \mathbb{F}_n \to \mathbb{P}^1$ be the ruling.
\begin{enumerate}
    \item \textbf{Picard Group:} The Picard group $\operatorname{Pic}(\mathbb{F}_n)$ is free abelian of rank 2, generated by the class of a fiber $f$ of $\pi$ and the class of a section $s$ satisfying $s^2 = -n$. Explicitly,
    $$\hlin{\operatorname{Pic}(\mathbb{F}_n) \cong \mathbb{Z} f \oplus \mathbb{Z} s,}$$
    where $f \cdot f = 0$, $f \cdot s = 1$, and $s$ corresponds to the quotient line bundle $\mathcal{O}_{\mathbb{P}^1}(n)$ of the defining vector bundle.
    \item \textbf{Unique Negative Section:} If $n > 0$, there exists a unique irreducible curve $C_0 \subset \mathbb{F}_n$ with negative self-intersection. This curve is a section of $\pi$ and satisfies $C_0^2 = -n$. It corresponds to the class $s$ in the Picard group.
    \item \textbf{Toric Description:} $\mathbb{F}_n$ is the complete toric surface associated to the fan $\Sigma_n \subset \mathbb{R}^2$ generated by the rays
    $$v_1 = (1, 0), \quad v_2 = (0, 1), \quad v_3 = (-1, n), \quad v_4 = (0, -1).$$
\end{enumerate}
\end{theorem}

\begin{proposition}[Isomorphism Classification]
Two Hirzebruch surfaces $\mathbb{F}_n$ and $\mathbb{F}_m$ are isomorphic as abstract algebraic varieties if and only if $n = m$.
\end{proposition}

\begin{theorem}[Rationality and Exceptional Cases]
\begin{itemize}
    \item $\mathbb{F}_0$ is isomorphic to $\mathbb{P}^1 \times \mathbb{P}^1$.
    \item $\mathbb{F}_1$ is isomorphic to the blow-up of $\mathbb{P}^2$ at a single point. The unique negative section $C_0$ (with $C_0^2 = -1$) is the exceptional divisor of the blow-up.
    \item All Hirzebruch surfaces $\mathbb{F}_n$ are rational surfaces.
\end{itemize}
\end{theorem}
